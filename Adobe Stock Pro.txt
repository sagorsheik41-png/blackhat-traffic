        // Keyword Research
        function generateKeywords() {
            const baseKeyword = document.getElementById('base-keyword').value.trim();
            const audience = document.getElementById('target-audience').value;
            
            if (!baseKeyword) {
                showNotification('Please enter a base keyword!', 'warning');

        function generateKeywordVariations(base, audience) {
            const prefixes = ['modern', 'professional', 'creative', 'innovative', 'digital', 'contemporary'];
            const suffixes = ['concept', 'design', 'illustration', 'vector', 'icon', 'graphic'];
            const modifiers = ['team', 'group', 'people', 'abstract', 'minimalist', 'detailed'];
            
            const audienceKeywords = {
                business: ['corporate', 'office', 'professional', 'executive'],
                creative: ['artistic', 'design', 'creative', 'innovative'],
                marketing: ['advertising', 'promotion', 'campaign', 'brand'],
                startup: ['entrepreneur', 'startup', 'innovation', 'growth'],
                corporate: ['enterprise', 'corporate', 'business', 'formal']
            };

            let keywords = [base];
            
            // Add prefix variations
            prefixes.forEach(prefix => {
                keywords.push(`${prefix} ${base}`);
            });
            
            // Add suffix variations
            suffixes.forEach(suffix => {
                keywords.push(`${base} ${suffix}`);
            });
            
            // Add modifier variations
            modifiers.forEach(modifier => {
                keywords.push(`${base} ${modifier}`);
            });
            
            // Add audience-specific keywords
            if (audienceKeywords[audience]) {
                audienceKeywords[audience].forEach(audienceKw => {
                    keywords.push(`${audienceKw} ${base}`);
                    keywords.push(`${base} ${audienceKw}`);
                });
            }

            return [...new Set(keywords)]; // Remove duplicates
        }

        function generateTitleSuggestions(original) {
            const powerWords = ['ultimate', 'complete', 'professional', 'premium', 'modern', 'creative', 'innovative', 'stunning'];
            const actionWords = ['boost', 'enhance', 'transform', 'elevate', 'optimize', 'maximize'];

            // Original with power words
            powerWords.forEach(word => {
                suggestions.push(`${word.charAt(0).toUpperCase() + word.slice(1)} ${original}`);

        // Utility Functions
        function copyToClipboard(text, element = null, isValue = false) {
            let textToCopy;
            
            if (isValue) {
                textToCopy = text;
            } else if (typeof text === 'string' && !element) {
                textToCopy = text;
            } else {
                const targetElement = typeof text === 'string' ? document.getElementById(text) : text;
                textToCopy = targetElement.value || targetElement.textContent;
            }
            
            navigator.clipboard.writeText(textToCopy).then(() => {
                showNotification('Copied to clipboard!');
                if (element && !isValue) {
                    element.textContent = 'Copied!';
                    setTimeout(() => {
                        element.textContent = 'Copy';
                    }, 1500);
                }
            }).catch(err => {
                console.error('Failed to copy: ', err);
                showNotification('Failed to copy to clipboard', 'error');
            });
        }

        function exportData() {
            const data = {
                stats: stats,
                activity: recentActivity,
                timestamp: new Date().toISOString()
            };
            
            const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `adobe_stock_data_${new Date().toISOString().slice(0, 10)}.json`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            showNotification('Data exported successfully!');
        }

        function importData() {
            const input = document.createElement('input');
            input.type = 'file';
            input.accept = '.json';
            input.onchange = function(event) {
                const file = event.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        try {
                            const data = JSON.parse(e.target.result);
                            if (data.stats) {
                                stats = data.stats;
                                updateStats();
                            }
                            if (data.activity) {
                                recentActivity = data.activity;
                                updateRecentActivity();
                            }
                            showNotification('Data imported successfully!');
                        } catch (error) {
                            showNotification('Error importing data!', 'error');
                        }
                    };
                    reader.readAsText(file);
                }
            };
            input.click();
        }
