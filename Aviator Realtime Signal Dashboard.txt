    // Simple single-file realtime signal dashboard.
    // CONFIG: If you have a real websocket or HTTP endpoint, paste it in the WebSocket / API URL field.

    const wsUrlEl = document.getElementById('wsUrl');
    const tokenEl = document.getElementById('token');
    const connectBtn = document.getElementById('connect');
    const disconnectBtn = document.getElementById('disconnect');
    const connStatus = document.getElementById('connStatus');
    const signalsEl = document.getElementById('signals');
    const logEl = document.getElementById('log');
    const copyBtn = document.getElementById('copyLatest');
    const balanceEl = document.getElementById('balance');
    const pauseBtn = document.getElementById('pause');
    const clearBtn = document.getElementById('clear');

    let ws = null;
    let paused = false;
    let latestSignalText = '';

    function addLog(text){
      const d = new Date().toLocaleTimeString();
      logEl.innerHTML = `<div style="padding:6px 8px;border-bottom:1px dashed rgba(255,255,255,0.02)"><strong>[${d}]</strong> ${escapeHtml(text)}</div>` + logEl.innerHTML;
    }

    function escapeHtml(s){return String(s).replace(/[&<>"']/g, c=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;' }[c]));}

    function setStatus(s,connected=false){
      connStatus.textContent = 'Status: ' + s;
      connStatus.className = connected? 'connected' : '';
    }

    function renderSignal(sig){
      // sig: {type:'signal',side:'cashout'|'stop',value:float,ts:int, note:string}
      const el = document.createElement('div');
      el.className = 'signal';
      const time = new Date((sig.ts||Date.now())*1000);
      const left = document.createElement('div'); left.className='sig-left';
      const badge = document.createElement('div'); badge.className='badge'; badge.textContent = sig.side?.toUpperCase() || (sig.type||'SIG');
      const text = document.createElement('div');
      text.innerHTML = `<div style="font-weight:700">${sig.note || (sig.side? sig.side + ' @ ' + (sig.value||'—'):'Signal')}</div><div class="sig-meta">${time.toLocaleTimeString()} • ${sig.source||'remote'}</div>`;
      left.appendChild(badge); left.appendChild(text);
      const right = document.createElement('div');
      right.style.display='flex'; right.style.flexDirection='column'; right.style.alignItems='flex-end';
      const copySmall = document.createElement('button'); copySmall.className='ghost'; copySmall.textContent='Copy';
      copySmall.onclick = ()=>{ navigator.clipboard.writeText(formatCopy(sig)); addLog('Copied signal to clipboard'); };
      right.appendChild(copySmall);
      el.appendChild(left); el.appendChild(right);
      signalsEl.prepend(el);
      // keep only last 60
      while(signalsEl.children.length>60) signalsEl.removeChild(signalsEl.lastChild);
    }

    function formatCopy(sig){
      // Single-line paste-friendly string
      const v = sig.value? (typeof sig.value==='number'? sig.value.toFixed(2): sig.value) : '';
      const t = sig.side? sig.side.toUpperCase() : (sig.type||'SIGNAL');
      const note = sig.note? ' — ' + sig.note : '';
      return `${t} ${v}${note}`;
    }

    copyBtn.addEventListener('click', ()=>{
      if(!latestSignalText){ addLog('No signal to copy'); return; }
      navigator.clipboard.writeText(latestSignalText).then(()=> addLog('Latest signal copied'));
    });

    pauseBtn.addEventListener('click', ()=>{
      paused = !paused; pauseBtn.textContent = paused? 'Resume' : 'Pause'; addLog(paused? 'Paused' : 'Resumed');
    });

    clearBtn.addEventListener('click', ()=>{signalsEl.innerHTML=''; addLog('Cleared signals');});

    connectBtn.addEventListener('click', ()=>{
      const url = wsUrlEl.value.trim();
      if(!url){alert('WebSocket/HTTP URL দিন (wss:// or https://)'); return;}
      // If WebSocket URL
      if(url.startsWith('ws')){
        connectWS(url);
      } else {
        // fallback: poll HTTP for new signals every 3s
        connectHTTPPoll(url);
      }
    });

    disconnectBtn.addEventListener('click', ()=>{ closeConnection(); addLog('Manual disconnect'); setStatus('Disconnected'); });

    // --- WebSocket connection ---
    function connectWS(url){
      closeConnection();
      try{
        const headersToken = tokenEl.value.trim();
        addLog('Connecting to ' + url);
        setStatus('Connecting...');
        // Native WebSocket can't set custom headers in browser; include token as query param if needed
        let finalUrl = url;
        if(headersToken){
          const sep = url.includes('?')? '&' : '?';
          finalUrl = url + sep + 'token=' + encodeURIComponent(headersToken);
        }
        ws = new WebSocket(finalUrl);
        ws.onopen = ()=>{ setStatus('Connected', true); addLog('WebSocket open'); }
        ws.onmessage = (ev)=>{ handleIncoming(ev.data); }
        ws.onclose = ()=>{ setStatus('Disconnected'); addLog('WebSocket closed'); ws=null; }
        ws.onerror = (e)=>{ addLog('WebSocket error'); }
      } catch(err){ addLog('WS connect error: '+err.message); setStatus('Error'); }
    }

    // --- HTTP polling fallback ---
    let pollTimer = null;
    function connectHTTPPoll(url){
      closeConnection();
      addLog('Starting HTTP poll: ' + url);
      setStatus('Polling...');
      pollTimer = setInterval(async ()=>{
        try{
          const headers = {};
          if(tokenEl.value.trim()) headers['Authorization'] = tokenEl.value.trim();
          const r = await fetch(url, {headers});
          if(!r.ok){ addLog('HTTP error ' + r.status); return; }
          const data = await r.json();
          // Expect either array of signals or single signal
          if(Array.isArray(data)) data.forEach(d=>handleIncoming(JSON.stringify(d)));
          else handleIncoming(JSON.stringify(data));
        }catch(e){ addLog('Poll error: '+e.message); }
      },3000);
    }

    function closeConnection(){
      if(ws){ ws.close(); ws=null; }
      if(pollTimer){ clearInterval(pollTimer); pollTimer=null; }
      setStatus('Disconnected');
    }

    function handleIncoming(raw){
      if(paused) return;
      addLog('Raw: '+ (typeof raw==='string'? raw : '[binary]'));
      let obj = null;
      try{ obj = typeof raw==='string'? JSON.parse(raw) : raw; } catch(e){
        // If incoming is plain text like "SIGNAL:2.5", parse lightly
        const t = String(raw).trim();
        const m = t.match(/([A-Za-z]+)[:@ ]\s*([0-9.]+)?/);
        if(m) obj = {type:'signal', side:m[1], value: m[2]? parseFloat(m[2]): undefined, note: t, ts: Math.floor(Date.now()/1000)};
        else obj = {type:'raw', note: String(raw), ts: Math.floor(Date.now()/1000)};
      }

      // Normalize common shapes
      const sig = normalizeSignal(obj);
      latestSignalText = formatCopy(sig);
      renderSignal(sig);
      updateBalance(sig);
      // small animation on chart placeholder
      flashChart(sig);
    }

    function normalizeSignal(obj){
      const out = {type:obj.type||'signal', side: obj.side || obj.action || obj.signal || (obj.type==='loss'?'STOP':'') , value: obj.value || obj.multiplier || obj.rate || obj.cashout, ts: obj.ts || Math.floor((Date.now())/1000), note: obj.note || obj.msg || obj.message || undefined, source: obj.source || 'remote'};
      return out;
    }

    function updateBalance(sig){
      // Demo: if signal carries balance field, show it
      if(sig && sig.balance){ balanceEl.textContent = 'Balance: ' + sig.balance + ' BDT'; }
    }

    function flashChart(sig){
      const chart = document.getElementById('chart');
      chart.textContent = sig.side? `${sig.side.toUpperCase()} @ ${sig.value||'—'}` : (sig.note||'Signal');
      chart.style.transition = 'transform 140ms ease';
      chart.style.transform = 'scale(1.02)';
      setTimeout(()=>{ chart.style.transform = 'scale(1)'; },150);
    }

    // Quick demo generator when no connection
    setTimeout(()=>{
      addLog('Tip: খুলে "WebSocket / API URL" ফিল্ড-এ আপনার endpoint দিন অথবা "Demo WS" একটিতে ক্লিক করুন।');
    },600);

    // Demo preset buttons
    document.querySelectorAll('button[data-url]').forEach(b=>b.addEventListener('click', ()=>{ wsUrlEl.value = b.dataset.url; addLog('Preset applied: ' + b.dataset.url); }));

    // Make the system produce demo signals if no connection for demoing
    let demoInterval = setInterval(()=>{
      if(ws || pollTimer) return;
      const sides = ['cashout','auto','stop','win'];
      const s = sides[Math.floor(Math.random()*sides.length)];
      const v = (Math.random()*15+1).toFixed(2);
      const obj = {type:'signal', side:s, value: parseFloat(v), ts: Math.floor(Date.now()/1000), note:'demo signal'};
      handleIncoming(JSON.stringify(obj));
    }, 9000);

  </script></body>
