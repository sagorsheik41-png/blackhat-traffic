  function createSlug(title) {

    if (hash) {

  // Helper function to update or create meta tags
  function updateOrCreateMeta(attr, attrValue, content) {
    let meta = document.querySelector(`meta[${attr}="${attrValue}"]`);
    if (!meta) {
      meta = document.createElement('meta');
      meta.setAttribute(attr, attrValue);
      document.head.appendChild(meta);
    }
    meta.setAttribute('content', content);
  }

  function handleScroll() {
    const scrolled = window.pageYOffset;
    
    if (scrolled > 200) {
      scrollToTopBtn.classList.add('visible');
    } else {
      scrollToTopBtn.classList.remove('visible');
    }

    // Infinite scrolling
    if (isLoading) return;
    
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    const windowHeight = window.innerHeight;
    const documentHeight = document.documentElement.scrollHeight;
    
    // Load more content when user is near the bottom (200px before)
    if (scrollTop + windowHeight >= documentHeight - 200) {
      const nextPage = currentPage + 1;
      if (currentType === 'movie' && !searchQuery) {
        loadMoviesByFilter(currentFilter, nextPage, true);
      } else if (currentType === 'tv' && !searchQuery) {
        loadTVShowsByFilter(currentFilter, nextPage, true);
      }
    }
  }

  // Person Details Modal Infinite Scroll Handler
  function handlePersonScroll() {
    if (isPersonLoading || !personDetailsModal.classList.contains('active')) return;
    
    const modalScrollTop = personDetailsModal.scrollTop;
    const modalScrollHeight = personDetailsModal.scrollHeight;
    const modalClientHeight = personDetailsModal.clientHeight;

  // Load more person credits
  function loadMorePersonCredits() {
    if (isPersonLoading || !allPersonCredits.length || displayedPersonCredits >= allPersonCredits.length) {
      // Show end message if all content is loaded
      if (displayedPersonCredits >= allPersonCredits.length && allPersonCredits.length > 0) {
        filmographyEnd.classList.add('show');
      }
      return;
    }
    
    isPersonLoading = true;
    filmographyLoading.classList.add('show');
    
    // Simulate loading delay for better UX
    setTimeout(() => {
      const startIndex = displayedPersonCredits;
      const endIndex = Math.min(startIndex + creditsPerPage, allPersonCredits.length);
      const creditsToShow = allPersonCredits.slice(startIndex, endIndex);
      
      // Append new credits to the grid
      creditsToShow.forEach(credit => {
        const isTV = credit.media_type === 'tv';
        const title = isTV ? credit.name : credit.title;
        const year = credit.release_date || credit.first_air_date;
        const yearText = year ? new Date(year).getFullYear() : 'TBA';
        
        const creditDiv = document.createElement('div');
        creditDiv.classList.add('filmography-item');
        creditDiv.dataset.id = credit.id;
        creditDiv.dataset.type = isTV ? 'tv' : 'movie';
        
        const posterImg = credit.poster_path 
          ? `https://image.tmdb.org/t/p/w300${credit.poster_path}` 
          : 'https://via.placeholder.com/300x450?text=No+Poster';
        
        let characterInfo = '';
        if (credit.credit_type === 'cast' && credit.character) {
          characterInfo = `<div class="filmography-character">as ${credit.character}</div>`;
        } else if (credit.credit_type === 'crew' && credit.job) {
          characterInfo = `<div class="filmography-character">${credit.job}</div>`;
        }
        
        creditDiv.innerHTML = `
          <img src="${posterImg}" alt="${title}" class="filmography-poster" loading="lazy" />
          <div class="filmography-title-text">${title}</div>
          <div class="filmography-year">${yearText}</div>
          ${characterInfo}
        `;
        
        creditDiv.addEventListener('click', () => {
          closePersonDetails();
          openMovieDetails(credit.id, isTV);
        });
        
        // Insert before loading indicator
        filmographyGrid.insertBefore(creditDiv, filmographyLoading);
      });
      
      displayedPersonCredits = endIndex;
      isPersonLoading = false;
      filmographyLoading.classList.remove('show');
      
      // Show end message if all content is loaded
      if (displayedPersonCredits >= allPersonCredits.length) {
        filmographyEnd.classList.add('show');
      }
      
    }, 800); // 800ms loading delay
  }

  // Fetch Popular Movies
  async function fetchPopularMovies(page = 1, append = false) {
    if (isLoading) return;
    isLoading = true;

    if (!append) {
      moviesContainer.innerHTML = "Loading popular movies...";
      currentPage = 1;
    }
    
    currentType = 'movie';

    try {
      const res = await fetch(`https://api.themoviedb.org/3/movie/popular?api_key=${apiKey}&language=en-US&page=${page}`);
      const data = await res.json();
      
      if (append) {
        appendItems(data.results);
      } else {
        displayItems(data.results);
      }
      
      currentPage = page;
      isLoading = false;
    } catch (e) {
      if (!append) {
        moviesContainer.innerHTML = "Failed to load popular movies.";
      }
      console.error(e);
      isLoading = false;
    }
  }

  // Search functionality
  async function searchMovies(query) {
    if (isLoading) return;
    isLoading = true;
    
    moviesContainer.innerHTML = "Searching...";
    searchQuery = query;
    
    try {
      const movieRes = fetch(`https://api.themoviedb.org/3/search/movie?api_key=${apiKey}&language=en-US&query=${encodeURIComponent(query)}&page=1`);
      const tvRes = fetch(`https://api.themoviedb.org/3/search/tv?api_key=${apiKey}&language=en-US&query=${encodeURIComponent(query)}&page=1`);
      
      const [movieData, tvData] = await Promise.all([movieRes, tvRes]);
      const movies = await movieData.json();
      const tvShows = await tvData.json();
      
      // Combine and sort results
      const allResults = [
        ...movies.results.map(item => ({...item, media_type: 'movie'})),
        ...tvShows.results.map(item => ({...item, media_type: 'tv'}))
      ].sort((a, b) => b.popularity - a.popularity);
      
      displaySearchResults(allResults);
      isLoading = false;
    } catch (e) {
      moviesContainer.innerHTML = "Search failed. Please try again.";
      console.error(e);
      isLoading = false;
    }
  }

  // Search for movie by slug when not found in localStorage
  async function searchMovieBySlug(slug, isTV) {
    try {
      // Convert slug back to a searchable title
      const searchTitle = slug.replace(/-/g, ' ');
      console.log('Searching for:', searchTitle);
      
      const searchUrl = isTV 
        ? `https://api.themoviedb.org/3/search/tv?api_key=${apiKey}&language=en-US&query=${encodeURIComponent(searchTitle)}`
        : `https://api.themoviedb.org/3/search/movie?api_key=${apiKey}&language=en-US&query=${encodeURIComponent(searchTitle)}`;
      
      const response = await fetch(searchUrl);
      const data = await response.json();
      
      if (data.results && data.results.length > 0) {
        // Find the best match by comparing slugs
        const bestMatch = data.results.find(item => {
          const itemTitle = isTV ? item.name : item.title;
          const itemSlug = createSlug(itemTitle);
          return itemSlug === slug;
        }) || data.results[0]; // Fallback to first result if exact match not found
        
        // Store the found data for future use
        storeMovieData(bestMatch.id, isTV ? bestMatch.name : bestMatch.title, isTV);
        
        // Open the movie details
        setTimeout(() => {
          openMovieDetails(bestMatch.id, isTV);
        }, 1000);
      } else {
        console.log('No search results found for slug:', slug);
      }
    } catch (error) {
      console.error('Error searching for movie by slug:', error);
    }
  }

  // Load all custom embed URLs on startup
  async function loadCustomEmbedUrls() {
    try {
      const [moviesSnapshot, tvSnapshot] = await Promise.all([
        database.ref('customEmbedUrls/movie').once('value'),
        database.ref('customEmbedUrls/tv').once('value')
      ]);
      
      const moviesData = moviesSnapshot.val() || {};
      const tvData = tvSnapshot.val() || {};
      
      // Load movie URLs
      Object.keys(moviesData).forEach(movieId => {
        if (moviesData[movieId] && moviesData[movieId].url) {
          customEmbedUrls[`movie_${movieId}`] = moviesData[movieId].url;
        }
      });
      
      // Load TV URLs
      Object.keys(tvData).forEach(tvId => {
        if (tvData[tvId] && tvData[tvId].url) {
          customEmbedUrls[`tv_${tvId}`] = tvData[tvId].url;
        }
      });
      
      // Update custom source button visibility
      updateCustomSourceVisibility();
      
    } catch (error) {
      console.error('Error loading custom embed URLs:', error);
    }
  }

  function showPersonDetails(person, credits) {
    // Reset person scroll variables
    currentPersonData = { person, credits };
    currentPersonPage = 1;
    isPersonLoading = false;
    displayedPersonCredits = 0;
    
    // Set person photo
    const imgBaseUrl = "https://image.tmdb.org/t/p/w500";
    personPhoto.src = person.profile_path ? imgBaseUrl + person.profile_path : 'https://via.placeholder.com/400x600?text=No+Photo';

    // Set person info
    personName.textContent = person.name || 'Unknown';
    personBio.textContent = person.biography || 'No biography available.';
    
    // Set personal details
    personBirthday.textContent = person.birthday ? new Date(person.birthday).toLocaleDateString('en-GB', { 
      day: '2-digit', 
      month: 'long', 
      year: 'numeric' 
    }) : 'N/A';
    
    personBirthplace.textContent = person.place_of_birth || 'N/A';
    personKnownFor.textContent = person.known_for_department || 'Acting';

    // Set filmography title
    filmographyTitle.textContent = `${person.known_for_department || 'Acting'} Credits`;

    // Combine and sort all credits by popularity and release date
    allPersonCredits = [
      ...credits.cast.map(item => ({...item, credit_type: 'cast'})),
      ...credits.crew.map(item => ({...item, credit_type: 'crew'}))
    ];

    // Remove duplicates and sort by release date (newest first) and popularity
    const uniqueCredits = allPersonCredits.reduce((acc, current) => {
      const existing = acc.find(item => item.id === current.id);
      if (!existing) {
        acc.push(current);
      }
      return acc;
    }, []);

    // Sort by release date (newest first) and popularity
    uniqueCredits.sort((a, b) => {
      const dateA = new Date(a.release_date || a.first_air_date || '1900-01-01');
      const dateB = new Date(b.release_date || b.first_air_date || '1900-01-01');
      
      if (dateB - dateA !== 0) return dateB - dateA;
      return (b.popularity || 0) - (a.popularity || 0);
    });

    allPersonCredits = uniqueCredits;

    // Clear previous filmography
    filmographyGrid.innerHTML = '';
    filmographyLoading.classList.remove('show');
    filmographyEnd.classList.remove('show');
    
    // Add loading and end indicators back to grid
    filmographyGrid.appendChild(filmographyLoading);
    filmographyGrid.appendChild(filmographyEnd);

    // Load initial credits (first batch)
    loadMorePersonCredits();

    // Show the modal
    personDetailsModal.classList.add('active');
    document.body.style.overflow = 'hidden';
    
    // Scroll to top of modal
    personDetailsModal.scrollTop = 0;
  }

    // Reset variables
    currentPersonData = null;

    isPersonLoading = false;
  }

  // Update video player with current source
  function updateVideoPlayer() {
    if (!currentMovieData) return;

    const moviePlayer = document.getElementById('moviePlayer');
    const { details, external, isTV } = currentMovieData;
    const source = videoSources[currentVideoSource];

    if (currentVideoSource === 6) {
      // Custom embed URL
      const movieId = details.id;
      const contentType = isTV ? 'tv' : 'movie';
      const customUrl = customEmbedUrls[`${contentType}_${movieId}`];
      
      if (customUrl) {
        moviePlayer.src = customUrl;
      } else {
        // If no custom URL, switch back to source 1
        currentVideoSource = 1;
        updateSourceSelector();
        updateVideoPlayer();
        return;
      }
    } else if (isTV) {
      const season = document.getElementById('seasonInput').value || 1;
      const episode = document.getElementById('episodeInput').value || 1;
      const imdbId = external.imdb_id || details.id;
      
      if (source.tv) {
        moviePlayer.src = source.tv(imdbId, season, episode);
      }
    } else {
      const imdbId = external.imdb_id || details.id;
      
      if (source.movie) {
        moviePlayer.src = source.movie(imdbId);
      }
    }
  }

  // Handle download button click
  async function handleDownloadClick() {
    if (!currentMovieData) return;
    
    // Check if button is disabled
    if (metadataDownloadBtn.classList.contains('disabled')) {
      return; // Do nothing if disabled
    }
    
    const movieId = currentMovieData.details.id;
    const isTV = currentMovieData.isTV;
    const contentType = isTV ? 'tv' : 'movie';
    
    try {
      // Check Firebase for download URL
      const urlSnapshot = await database.ref(`downloadUrls/${contentType}/${movieId}`).once('value');
      const urlData = urlSnapshot.val();
      
      if (urlData && urlData.url) {
        // Open URL in new tab
        window.open(urlData.url, '_blank');
      } else {
        // This shouldn't happen if checkDownloadUrl is working correctly
        console.log('No URL found but button not disabled');
      }
    } catch (error) {
      console.error('Error fetching download URL:', error);
    }
  }

  // Check if download URL exists for current movie
  async function checkDownloadUrl(movieId, isTV) {
    try {
      const contentType = isTV ? 'tv' : 'movie';
      const urlSnapshot = await database.ref(`downloadUrls/${contentType}/${movieId}`).once('value');
      const urlData = urlSnapshot.val();
      
      if (urlData && urlData.url) {
        // URL available - normal colorful button with text
        metadataDownloadBtn.classList.remove('disabled');
        metadataDownloadBtn.innerHTML = 'Download';
        metadataDownloadBtn.style.pointerEvents = 'auto';
      } else {
        // No URL - disabled white/black button
        metadataDownloadBtn.classList.add('disabled');
        metadataDownloadBtn.innerHTML = 'Download';
        metadataDownloadBtn.style.pointerEvents = 'none';
      }
    } catch (error) {
      console.error('Error checking download URL:', error);
      // On error, show disabled state
      metadataDownloadBtn.classList.add('disabled');
      metadataDownloadBtn.innerHTML = 'Download';
      metadataDownloadBtn.style.pointerEvents = 'none';
    }
  }
