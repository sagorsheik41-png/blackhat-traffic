    <script>
        // Global variable to hold the single generated link data
        let generatedLink = null;
        
        // **Important:** For implementation, you must use your own domain here.
        const MY_DOMAIN = "https://short.ly"; // This is an example. Use your own domain.

        // Generate a single, direct short link
        function generateShortLink() {
            const originalUrl = document.getElementById('originalUrl').value.trim();
            const customAlias = document.getElementById('customAlias').value.trim();
            const expiryDate = document.getElementById('expiryDate').value;

            if (!isValidUrl(originalUrl)) {
                showNotification('❌ Please enter a valid URL', 'error');
                return;
            }

            const shortCode = customAlias || generateShortCode();
            const shortUrl = `${MY_DOMAIN}/${shortCode}`;
            
            generatedLink = {
                shortUrl: shortUrl,
                originalUrl: originalUrl,
                created: new Date(),
                expiryDate: expiryDate,
                clicks: 0
            };
            
            displayResult();
            showNotification('✅ Shortlink created successfully!', 'success');
        }

        // Display the generated result
        function displayResult() {
            const resultsSection = document.getElementById('resultsSection');
            resultsSection.classList.remove('hidden');

            // 1. Create and display the link card
            const shortLinkCardContainer = document.getElementById('shortLinkCardContainer');
            shortLinkCardContainer.innerHTML = createLinkCard(generatedLink);
            
            // 2. Generate and display QR code
            generateQRCode(generatedLink.shortUrl);
            
            // 3. Show link preview
            showLinkPreview(generatedLink.originalUrl);
        }

        // Create the main link card
        function createLinkCard(link) {
            return `
                <div class="bg-white rounded-xl p-6 shadow-lg h-full flex flex-col justify-between">
                    <div>
                        <div class="flex items-center justify-between mb-4">
                             <h4 class="font-semibold text-xl text-gray-800">Your Shortlink</h4>
                            <div class="text-right">
                                <div class="text-sm text-gray-500">Clicks</div>
                                <div id="click-counter" class="text-2xl font-bold text-blue-600">${link.clicks}</div>
                            </div>
                        </div>
                        
                        <p class="text-sm text-gray-600 mb-2">Original Link:</p>
                        <p class="text-xs text-gray-500 bg-gray-100 p-2 rounded break-all mb-4">${link.originalUrl}</p>
                        
                        <div class="bg-gray-50 rounded-lg p-4 mb-4">
                            <label class="text-sm text-gray-600">Short Link:</label>
                            <div class="flex items-center justify-between mt-1">
                                <input 
                                    type="text" 
                                    value="${link.shortUrl}" 
                                    readonly 
                                    id="shortUrlInput"
                                    class="bg-transparent flex-1 text-gray-900 font-mono text-lg"
                                />
                                <button 
                                    onclick="copyToClipboard('shortUrlInput')" 
                                    class="ml-2 px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-colors"
                                >
                                    <i class="fas fa-copy"></i>
                                </button>
                            </div>
                        </div>
                    </div>
                    
                    <div class="flex space-x-2 mt-4">
                        <a href="${link.originalUrl}" target="_blank" onclick="trackClick()" class="w-full text-center px-4 py-3 bg-green-600 text-white rounded-lg hover:bg-green-700 transition-colors">
                            <i class="fas fa-external-link-alt mr-2"></i>Test the Link
                        </a>
                    </div>
                    
                    ${link.expiryDate ? `<div class="mt-3 text-xs text-center text-orange-600"><i class="fas fa-clock mr-1"></i>Expires on: ${new Date(link.expiryDate).toLocaleDateString('en-US')}</div>` : ''}
                </div>
            `;
        }
        
        // Function to track clicks
        function trackClick() {
            if (generatedLink) {
                generatedLink.clicks++;
                document.getElementById('click-counter').textContent = generatedLink.clicks;
            }
        }

        // Generate a single QR code
        function generateQRCode(url) {
            const qrContainer = document.getElementById('qrCodeContainer');
            qrContainer.innerHTML = ''; // Clear previous QR code
            
            const canvas = document.createElement('canvas');
            canvas.id = 'qrCanvas';
            qrContainer.appendChild(canvas);

            QRCode.toCanvas(canvas, url, {
                width: 150,
                height: 150,
                color: {
                    dark: '#1f2937',
                    light: '#ffffff'
                }
            });
        }

        // Show link preview
        function showLinkPreview(url) {
            const linkPreview = document.getElementById('linkPreview');
            linkPreview.innerHTML = `
                <div class="flex items-center space-x-4">
                    <div class="w-12 h-12 bg-gray-200 rounded-lg flex items-center justify-center">
                        <i class="fas fa-globe text-gray-500 text-2xl"></i>
                    </div>
                    <div class="flex-1">
                        <p class="text-gray-600 text-sm break-all">${url}</p>
                        <p class="text-xs text-gray-500 mt-1">This link will take the user directly here.</p>
                    </div>
                </div>
            `;
        }

        // Utility functions
        function copyToClipboard(elementId) {
            const element = document.getElementById(elementId);
            element.select();
            element.setSelectionRange(0, 99999);
            document.execCommand('copy');
            showNotification('📋 Link copied!', 'success');
        }
        
        function downloadQR() {
            const canvas = document.getElementById('qrCanvas');
            const link = document.createElement('a');
            link.download = 'qrcode.png';
            link.href = canvas.toDataURL();
            link.click();
            showNotification('📥 Downloading QR Code...', 'success');
        }

        function showNotification(message, type = 'success') {
            const notification = document.createElement('div');
            notification.className = `copy-success ${type === 'error' ? 'bg-red-500' : 'bg-green-500'}`;
            notification.textContent = message;
            document.body.appendChild(notification);
            setTimeout(() => { notification.remove(); }, 3000);
        }

        function generateShortCode() {
            return Math.random().toString(36).substring(2, 8);
        }

        function isValidUrl(string) {
            try {
                new URL(string);
                return string.startsWith('http://') || string.startsWith('https://');
            } catch (_) {
                return false;
            }
        }
