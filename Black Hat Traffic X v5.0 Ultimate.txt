        function startTrafficGeneration() {
            const targetUrl = document.getElementById('target-url').value.trim();
            const visitCount = parseInt(document.getElementById('visit-count').value);
            const delay = parseInt(document.getElementById('delay-seconds').value) * 1000;
            const threads = parseInt(document.getElementById('thread-count').value);

            if (!targetUrl || !isValidUrl(targetUrl)) {
                showAppNotification('Please enter a valid Target URL.', 'error');
                return;
            }
            if (isNaN(visitCount) || visitCount <= 0) {
                showAppNotification('Please enter a valid number of visits.', 'error');
                return;
            }
            if (isNaN(delay) || delay < 0) {
                showAppNotification('Please enter a valid delay.', 'error');
                return;
            }
            if (isNaN(threads) || threads <= 0 || threads > 20) {
                showAppNotification('Threads must be between 1 and 20.', 'error');
                return;
            }

            AppState.trafficStats.isRunning = true;
            AppState.trafficStats.startTime = Date.now();
            AppState.trafficStats.successfulVisits = 0;
            AppState.trafficStats.failedVisits = 0;

            document.getElementById('start-traffic').disabled = true;
            document.getElementById('stop-traffic').disabled = false;
            document.getElementById('traffic-status').textContent = 'Running';

            addTrafficLog('Traffic generation started.', 'success');
            showAppNotification('Traffic generation process initiated!', 'success');

            for (let i = 0; i < threads; i++) {
                setTimeout(() => runTrafficSimulation(targetUrl, visitCount, delay, i), i * 100);
            }

            AppState.intervals.timer = setInterval(updateTrafficTimer, 1000);
            updateTrafficProgress();
        }

        function stopTrafficGeneration() {
            AppState.trafficStats.isRunning = false;
            clearInterval(AppState.intervals.timer);

            document.getElementById('start-traffic').disabled = false;
            document.getElementById('stop-traffic').disabled = true;
            document.getElementById('traffic-status').textContent = 'Stopped';

            addTrafficLog('Traffic generation stopped by user.', 'warning');
            showAppNotification('Traffic generation stopped!', 'warning');
        }

        function runTrafficSimulation(url, totalVisits, delay, threadId) {
            if (!AppState.trafficStats.isRunning) return;

            const visitsPerThread = Math.ceil(totalVisits / parseInt(document.getElementById('thread-count').value));
            let visitsDoneByThisThread = 0;

            async function sendVisit() {
                if (!AppState.trafficStats.isRunning || visitsDoneByThisThread >= visitsPerThread || 
                    (AppState.trafficStats.successfulVisits + AppState.trafficStats.failedVisits) >= totalVisits) {
                    if ((AppState.trafficStats.successfulVisits + AppState.trafficStats.failedVisits) >= totalVisits && AppState.trafficStats.isRunning) {
                        stopTrafficGeneration();
                    }
                    return;
                }

                const isSuccess = Math.random() > 0.15;
                if (isSuccess) {
                    AppState.trafficStats.successfulVisits++;
                    AppState.trafficStats.totalVisits++;
                    AppState.trafficStats.todayVisits++;
                    addTrafficLog(`Visit to ${url.substring(0,30)}... OK (Thread-${threadId + 1})`, 'success');
                } else {
                    AppState.trafficStats.failedVisits++;
                    addTrafficLog(`Visit to ${url.substring(0,30)}... FAILED (Thread-${threadId + 1})`, 'error');
                }

                visitsDoneByThisThread++;
                updateTrafficProgress();
                updateDashboardStats();

                if (AppState.trafficStats.isRunning) {
                    setTimeout(sendVisit, delay + Math.random() * 500);
                }
            }

            sendVisit();
        }

        function updateTrafficProgress() {
            const processed = AppState.trafficStats.successfulVisits + AppState.trafficStats.failedVisits;
            const total = parseInt(document.getElementById('visit-count').value);
            const progress = total > 0 ? Math.min((processed / total) * 100, 100) : 0;
            
            const progressBar = document.getElementById('traffic-progress');
            progressBar.style.width = `${progress}%`;
            progressBar.textContent = `${Math.round(progress)}%`;

            document.getElementById('successful-visits').textContent = AppState.trafficStats.successfulVisits.toLocaleString();
            document.getElementById('failed-visits').textContent = AppState.trafficStats.failedVisits.toLocaleString();
            
            if (AppState.settings.autoSave) saveToStorage('trafficStats', AppState.trafficStats);
        }

        function updateTrafficTimer() {
            if (!AppState.trafficStats.isRunning || !AppState.trafficStats.startTime) return;
            const elapsed = Date.now() - AppState.trafficStats.startTime;
            const h = String(Math.floor(elapsed / 3600000)).padStart(2, '0');
            const m = String(Math.floor((elapsed % 3600000) / 60000)).padStart(2, '0');
            const s = String(Math.floor((elapsed % 60000) / 1000)).padStart(2, '0');
            document.getElementById('running-time').textContent = `${h}:${m}:${s}`;
        }

        async function callGeminiAPI(message) {
            const API_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash-latest:generateContent?key=${AppState.settings.geminiApiKey}`;
            const response = await fetch(API_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ contents: [{ parts: [{ text: message }] }] })
            });

            if (!response.ok) {
                const errorData = await response.json();
                throw new Error(errorData.error?.message || `API request failed with status ${response.status}`);
            }

            const data = await response.json();
            let text = data.candidates?.[0]?.content?.parts?.[0]?.text || null;
            if (text) {
                text = text.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                           .replace(/\*(.*?)\*/g, '<em>$1</em>')
                           .replace(/```([\s\S]*?)```/g, '<pre class="bg-gray-800 p-2 rounded text-sm my-2 whitespace-pre-wrap"><code>$1</code></pre>')
                           .replace(/`([^`]+)`/g, '<code>$1</code>')
                           .replace(/(?:\r\n|\r|\n)/g, '<br>');}
            return text;
        }

        async function handleFileProcessing() {
            const filesInput = document.getElementById('files-input');
            const folderInput = document.getElementById('folder-input');
            let filesToProcess = [];

            if (filesInput.files.length > 0) {
                filesToProcess = Array.from(filesInput.files);
            } else if (folderInput.files.length > 0) {
                filesToProcess = Array.from(folderInput.files);
            }
            
            const supportedExtensions = ['.svg', '.png', '.jpg', '.jpeg', '.psd', '.eps', '.ai', '.webp'];
            filesToProcess = filesToProcess.filter(file => 
                supportedExtensions.some(ext => file.name.toLowerCase().endsWith(ext))
            );

            if (filesToProcess.length === 0) {
                showAppNotification('No supported files found. Please select SVG, PNG, JPG, PSD, EPS, AI, or WEBP files.', 'error');
                return;
            }

            const outputDisplay = document.getElementById('file-output-display');
            outputDisplay.innerHTML = `<div class="text-center py-12"><i class="loading-spinner"></i> <span class="ml-4 text-lg font-medium">Processing ${filesToProcess.length} file(s)...</span></div>`;
            AppState.processedFiles = [];

            for (let i = 0; i < filesToProcess.length; i++) {
                try {
                    const processedData = await processSingleFile(filesToProcess[i]);
                    AppState.processedFiles.push(processedData);
                    updateFileOutputDisplay(processedData, i + 1, filesToProcess.length);
                    if (i % 5 === 0 && outputDisplay.querySelector('span.ml-4.text-lg.font-medium')) {
                         outputDisplay.querySelector('span.ml-4.text-lg.font-medium').textContent = `Processing ${filesToProcess.length} file(s)... (${i+1}/${filesToProcess.length})`;
                    }
                } catch (error) {
                    console.error('File processing error:', filesToProcess[i].name, error);
                    updateFileOutputDisplay({ 
                        originalName: filesToProcess[i].name, 
                        error: error.message, 
                        fileType: filesToProcess[i].name.split('.').pop().toLowerCase() 
                    }, i + 1, filesToProcess.length, true);
                }
            }
            
            const successfulFiles = AppState.processedFiles.filter(f => !f.error);
            if(successfulFiles.length > 0) {
                document.getElementById('download-all-files-btn').disabled = false;
                document.getElementById('download-csv-btn').disabled = false;
                showAppNotification(`${successfulFiles.length} file(s) processed successfully!`, 'success');
            } else if (filesToProcess.length > 0) {
                showAppNotification(`All ${filesToProcess.length} files failed to process. Check console for errors.`, 'error');
                 outputDisplay.innerHTML = `<p style="color: var(--danger-color-start); text-align: center; padding: 50px 0;">All files failed processing. Please check console for errors.</p>`;
            }


            updateDashboardStats();
            if (AppState.settings.autoSave) saveToStorage('processedFiles', AppState.processedFiles);
        }

        async function processSingleFile(file) {
            const originalName = file.name;
            const fileExtension = originalName.split('.').pop().toLowerCase();
            const baseNameForTitle = cleanBaseFileName(originalName); 

            const adobeTitle = generateAdobeStockTitle(baseNameForTitle, fileExtension);
            const adobeKeywords = generateAdobeStockKeywords(adobeTitle, baseNameForTitle, fileExtension);
            const adobeDescription = generateAdobeStockDescription(adobeTitle, baseNameForTitle, fileExtension);
            
            const descriptiveFileName = `${adobeTitle}.${fileExtension}`; 
            const seoSlugName = generateSEOFileName(adobeTitle, fileExtension);

            let fileDataForZip;
            let previewUrl = null;

            if (fileExtension === 'svg') {
                const svgTextContent = await readFileAsText(file);
                const modifiedSvgText = modifySVGContent(svgTextContent, adobeTitle, adobeDescription);
                fileDataForZip = modifiedSvgText; 
                try { previewUrl = `data:image/svg+xml;base64,${btoa(unescape(encodeURIComponent(modifiedSvgText)))}`; } 
                catch(e) { console.warn("SVG preview gen error for " + originalName, e); }
            } else if (['png', 'jpg', 'jpeg', 'webp'].includes(fileExtension)) {
                fileDataForZip = file; 
                try { previewUrl = URL.createObjectURL(file); }
                catch(e) { console.warn("Image preview gen error for " + originalName, e); }
            } else {
                fileDataForZip = file; 
            }
            
            return {
                id: `file-${Date.now()}-${Math.random().toString(16).slice(2)}`,
                originalName,
                dialogFileName: descriptiveFileName, 
                newName: seoSlugName, 
                title: adobeTitle,
                keywords: adobeKeywords.join(', '),
                description: adobeDescription,
                fileType: fileExtension.toUpperCase(),
                fileSize: formatFileSize(file.size),
                originalFileObject: file, 
                fileDataForZip: fileDataForZip, 
                previewUrl: previewUrl,
                processedAt: new Date().toISOString()
            };
        }
        
        function cleanBaseFileName(fileName){
            return fileName.substring(0, fileName.lastIndexOf('.'))
                           .replace(/[\s._-]+/g, ' ') 
                           .replace(/[^a-zA-Z0-9 ]/g, '') 
                           .replace(/\s+/g, ' ') 
                           .trim()
                           .split(' ')
                           .map(word => word.charAt(0).toUpperCase() + word.slice(1))
                           .join(' ') || "Untitled_Asset";
        }

        function generateSEOFileName(generatedTitle, fileExtension) { 
            let fileName = generatedTitle.toLowerCase()
                .replace(/&/g, 'and') 
                .replace(/[^a-z0-9\s-]/g, '') 
                .trim()
                .replace(/\s+/g, '-') 
                .replace(/-+/g, '-') 
                .replace(/^-+|-+$/g, ''); 

            const maxLength = 70; 
            if (fileName.length > maxLength) {
                let tempName = fileName.substring(0, maxLength);
                const lastHyphen = tempName.lastIndexOf('-');
                if (lastHyphen > Math.floor(maxLength * 0.5) && lastHyphen > 10) { 
                    fileName = tempName.substring(0, lastHyphen);
                } else {
                    fileName = tempName;
                }
                fileName = fileName.replace(/-+$/, ''); 
            }
            
            if (!fileName || fileName.length < 5) {
                const fallbackBase = generatedTitle.toLowerCase().replace(/[^a-z0-9]/g, '').substring(0,20) || "asset";
                fileName = `${fallbackBase}-${Date.now().toString().slice(-6)}`;
            }
            
            return `${fileName}.${fileExtension.toLowerCase()}`;
        }
        
        function readFileAsText(file){
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = e => resolve(e.target.result);
                reader.onerror = e => reject(new Error("Error reading file: " + e.target.error.name));
                reader.readAsText(file);
            });
        }
        
        function modifySVGContent(svgContent, title, description){
            try {
                const parser = new DOMParser();
                const doc = parser.parseFromString(svgContent, 'image/svg+xml');
                const svgElement = doc.documentElement;

                if (svgElement.nodeName.toLowerCase() !== 'svg' || doc.getElementsByTagName("parsererror").length > 0) {
                    console.warn("Parsed document is not a valid SVG. Returning original content.");
                    return svgContent; 
                }
        
                let titleElement = svgElement.querySelector('title');
                if (!titleElement) {
                    titleElement = doc.createElementNS('http://www.w3.org/2000/svg', 'title');
                    svgElement.insertBefore(titleElement, svgElement.firstChild);
                }
                titleElement.textContent = title;
        
                let descElement = svgElement.querySelector('desc');
                if (!descElement) {
                    descElement = doc.createElementNS('http://www.w3.org/2000/svg', 'desc');
                    svgElement.insertBefore(descElement, titleElement.nextSibling || svgElement.firstChild);
                }
                descElement.textContent = description;
        
                return new XMLSerializer().serializeToString(doc);
            } catch (e) {
                console.error("Error modifying SVG content:", e);
                return svgContent; 
            }
        }

        function formatFileSize(bytes, decimals = 2){
            if (bytes === 0) return '0 Bytes';
            const k = 1024;
            const dm = decimals < 0 ? 0 : decimals;
            const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
        }

        async function downloadSingleProcessedFile(fileId){
            const fileData = AppState.processedFiles.find(f => f.id === fileId);
            if (!fileData || fileData.error) {
                showAppNotification('File data not found or file has an error.', 'error');
                return;
            }

            let blob;
            if (fileData.fileType === 'SVG') {
                blob = new Blob([fileData.fileDataForZip], { type: 'image/svg+xml' });
            } else {
                blob = new Blob([await fileData.originalFileObject.arrayBuffer()], { type: fileData.originalFileObject.type });
            }
            
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = fileData.dialogFileName; 
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            showAppNotification(`Downloading: ${DOMPurify.sanitize(fileData.dialogFileName)}`, 'success');
        }

        async function downloadAllProcessedFiles(){
            const validFiles = AppState.processedFiles.filter(f => !f.error);
            if (validFiles.length === 0) {
                showAppNotification('No successfully processed files to download.', 'error');
                return;
            }

            const zip = new JSZip();
            validFiles.forEach(fileData => {
                zip.file(fileData.dialogFileName, fileData.fileDataForZip); 
            });

            try {
                const content = await zip.generateAsync({ type: 'blob' });
                const url = URL.createObjectURL(content);
                const a = document.createElement('a');
                a.href = url;
                a.download = `adobe-stock-batch-${Date.now()}.zip`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                showAppNotification(`${validFiles.length} file(s) ZIP download started! Files inside will use descriptive titles.`, 'success');
            } catch (error) {
                console.error('ZIP creation error:', error);
                showAppNotification('Error creating ZIP file. ' + error.message, 'error');
            }
        }
        
        function downloadProcessedFilesCSV(){
            const validFiles = AppState.processedFiles.filter(f => !f.error);
            if (validFiles.length === 0) {
                showAppNotification('No data to create CSV from successfully processed files.', 'error');
                return;
            }

            // Adobe Stock CSV Headers - CRITICAL for successful upload
            const headers = ['Filename', 'Title', 'Keywords', 'Description']; // Added Description

            const csvRows = validFiles.map(file => {
                const rowData = [
                    file.dialogFileName,       // This MUST match the filename uploaded to Adobe Stock
                    file.title,
                    file.keywords,
                    file.description           // Added Description field
                ];
                // Ensure all fields are properly quoted and internal quotes are escaped
                return rowData.map(field => `"${String(field == null ? '' : field).replace(/"/g, '""')}"`).join(',');
            });

            const csvContent = [headers.join(','), ...csvRows].join('\n');

            const blob = new Blob([new Uint8Array([0xEF, 0xBB, 0xBF]), csvContent], { type: 'text/csv;charset=utf-8;' }); 
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `adobe_stock_metadata_${Date.now()}.csv`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            showAppNotification('Adobe Stock compatible CSV download started!', 'success');
        }

        function generateUniqueNewContent(count = 6) {
            const contents = [];
            const { selectedContentType, contentTypes, styles, types, purposes, emotions, industries } = AppState.newContentGenerator;
            const subjectsPool = contentTypes[selectedContentType]?.subjects || contentTypes.mixed.subjects;

            if (!subjectsPool || subjectsPool.length === 0) {
                console.error(`No subjects for content type: ${selectedContentType}`);
                return [];
            }
        
            let attempts = 0;
            const maxAttemptsPerItem = 50; 
            let totalAttempts = 0;
            const maxTotalAttemptsOverall = count * maxAttemptsPerItem * 2; 


            while (contents.length < count && totalAttempts < maxTotalAttemptsOverall) {
                totalAttempts++;
                let currentItemAttempts = 0;
                let foundUnique = false;

                while(!foundUnique && currentItemAttempts < maxAttemptsPerItem && totalAttempts < maxTotalAttemptsOverall) {
                    totalAttempts++;
                    currentItemAttempts++;

                    const subject = subjectsPool[Math.floor(Math.random() * subjectsPool.length)];
                    const style = styles[Math.floor(Math.random() * styles.length)];
                    const type = types[Math.floor(Math.random() * types.length)];
                    const purpose = purposes[Math.floor(Math.random() * purposes.length)];
                    const emotion = emotions[Math.floor(Math.random() * emotions.length)];
                    const industry = industries[Math.floor(Math.random() * industries.length)];
                    
                    const combinationKey = `${selectedContentType}-${subject}-${style}-${type}-${purpose}-${emotion}-${industry}`.toLowerCase().replace(/\s+/g, '-');
                    
                    const isUsed = AppState.generatedAIContent.some(c => c.combinationKey === combinationKey);
                    if (!isUsed) {
                        const content = createNewContentFromCombination({ subject, style, type, purpose, emotion, industry, combinationKey, selectedContentType }, contents.length + 1);
                        contents.push(content);
                        AppState.generatedAIContent.push({ combinationKey, timestamp: Date.now() });
                        if(AppState.generatedAIContent.length > 10000) { 
                            AppState.generatedAIContent.sort((a,b) => a.timestamp - b.timestamp).splice(0, AppState.generatedAIContent.length - 10000);
                        }
                        foundUnique = true;
                    }
                }
                if (!foundUnique && contents.length < count) {
                } else if (!foundUnique) {
                    break; 
                }
            }

            if (contents.length < count) {
                 showAppNotification(`Generated ${contents.length} items. Unique combinations might be limited for '${selectedContentType}'. Try another focus or clear history.`, 'warning');
            }
            if(AppState.settings.autoSave) saveToStorage('generatedAIContent', AppState.generatedAIContent);
            return contents;
        }

        function copyToClipboard(text, button, type = 'Text') {
            navigator.clipboard.writeText(text).then(() => {
                const originalHTML = button.innerHTML;
                button.innerHTML = '<i class="fas fa-check mr-1"></i>Copied!';
                button.classList.add('!bg-green-500'); 
                setTimeout(() => {
                    button.innerHTML = originalHTML;
                    button.classList.remove('!bg-green-500');
                }, 2000);
                showAppNotification(`${type} copied to clipboard!`, 'success');
            }).catch(err => {
                console.error('Copy failed:', err);
                showAppNotification(`Error copying ${type.toLowerCase()}. Check console.`, 'error');
            });
        }

        function addAdobeFileToTrack() {
            const fileName = document.getElementById('adobe-file-name').value.trim();
            const uploadDate = document.getElementById('adobe-upload-date').value;
            const status = document.getElementById('adobe-file-status').value;
            const downloadCount = parseInt(document.getElementById('adobe-download-count').value) || 0;

            if (!fileName || !uploadDate) {
                showAppNotification('Please enter asset title and upload date.', 'error');
                return;
            }

            const newFile = {
                id: `adobestock-${Date.now()}`,
                name: fileName,
                uploadDate,
                status,
                downloads: downloadCount,
                estimatedRevenue: calculateAdobeEstimatedRevenue(downloadCount, status),
                addedAt: new Date().toISOString()
            };

            AppState.adobeStockFiles.push(newFile);
            AppState.adobeStockFiles.sort((a,b) => new Date(b.uploadDate) - new Date(a.uploadDate));
            saveToStorage('adobeStockFiles', AppState.adobeStockFiles);
            
            document.getElementById('adobe-file-name').value = '';
            document.getElementById('adobe-download-count').value = '0';

            renderAdobeTrackedFiles();
            updateAdobeStatsDisplay();
            showAppNotification('Asset added to tracker!', 'success');
        }

        function calculateAdobeEstimatedRevenue(downloads, status) {
             if (status !== 'Approved' && status !== 'Live & Selling') return 0; 
            const avgRevenuePerDownload = 0.33; 
            return downloads * avgRevenuePerDownload;
        }

        function generateAdobeKeywordsFromConcept() {
            const description = document.getElementById('keyword-content-description').value.trim();
            const category = document.getElementById('keyword-category').value;

            if (!description) {
                showAppNotification('Please describe your content concept for keyword generation.', 'error');
                return;
            }

            let keywords = new Set();
            description.toLowerCase().match(/\b\w{3,}\b/g)?.forEach(word => keywords.add(word));
            keywords.add(category); 

            const commonStockKeywords = ['illustration', 'vector', 'design', 'graphic', 'background', 'abstract', 'modern', 'creative', 'template', 'business', 'technology', 'concept', 'art', 'wallpaper', 'banner', 'poster', 'icon', 'symbol', 'element'];
            commonStockKeywords.forEach(k => keywords.add(k));

            const catSpecific = {business: ['corporate', 'finance', 'office'], technology: ['digital', 'futuristic', 'data']};
            (catSpecific[category] || []).forEach(k => keywords.add(k));

            let finalKeywords = Array.from(keywords);
            finalKeywords.sort(() => 0.5 - Math.random()); 
            finalKeywords = finalKeywords.slice(0, 50); 

            displayAdobeKeywords(finalKeywords);
            showAppNotification(`Generated ${finalKeywords.length} optimized keywords!`, 'success');
        }

        function displayAdobeKeywords(keywords) {
            const container = document.getElementById('keyword-container');
            const resultsDiv = document.getElementById('keyword-results');
            const countSpan = document.getElementById('keyword-count');

            container.innerHTML = keywords.map(keyword => `<span class="keyword-tag">${DOMPurify.sanitize(keyword)}</span>`).join('');
            countSpan.textContent = keywords.length;
            resultsDiv.style.display = 'block';
            window.currentAdobeKeywords = keywords; 
        }

        function copyGeneratedAdobeKeywords() {
            if (!window.currentAdobeKeywords || window.currentAdobeKeywords.length === 0) {
                showAppNotification('Generate keywords first.', 'error');
                return;
            }
            copyToClipboard(window.currentAdobeKeywords.join(', '), document.getElementById('copy-keywords-btn'), 'Keywords');
        }

        function exportGeneratedAdobeKeywordsCSV() {
            if (!window.currentAdobeKeywords || window.currentAdobeKeywords.length === 0) {
                showAppNotification('Generate keywords first.', 'error');
                return;
            }
            const csvContent = [
                ['Keyword'], 
                ...window.currentAdobeKeywords.map(keyword => [keyword]) 
            ].map(row => row.join(',')).join('\n');

            const blob = new Blob([new Uint8Array([0xEF, 0xBB, 0xBF]), csvContent], { type: 'text/csv;charset=utf-8;' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `adobe_stock_keywords_${Date.now()}.csv`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            showAppNotification('Keywords exported as CSV!', 'success');
        }

        function addSocialLink() {
            const nameInput = document.getElementById('social-link-name');
            const urlInput = document.getElementById('social-link-url');
            const platformSelect = document.getElementById('social-link-platform');

            const name = nameInput.value.trim();
            const url = urlInput.value.trim();
            const platform = platformSelect.value;

            if (!name || !url || !isValidUrl(url)) {
                showAppNotification('Please enter a valid name and URL for the social link.', 'error');
                return;
            }

            const newLink = {
                id: `social-${Date.now()}`,
                name, url, platform,
                visits: 0,
                lastClick: null,
                addedAt: new Date().toISOString()
            };
            AppState.socialLinks.push(newLink);
            AppState.socialLinks.sort((a,b) => a.name.localeCompare(b.name));
            saveToStorage('socialLinks', AppState.socialLinks);

            nameInput.value = '';
            urlInput.value = '';
            renderSocialLinksTable();
            showAppNotification('Social link added successfully!', 'success');
            updateDashboardStats();
        }

        function renderSocialLinksTable() {
            const tbody = document.getElementById('links-table-body');
            if (AppState.socialLinks.length === 0) {
                tbody.innerHTML = `<tr><td colspan="5" class="text-center text-gray-400 py-10">No social links tracked yet.</td></tr>`;
                return;
            }
            tbody.innerHTML = AppState.socialLinks.map(link => {
                const platformDetails = getPlatformDetails(link.platform);
                return `
                    <tr>
                        <td><a href="${link.url}" target="_blank" onclick="trackSocialLinkClick('${link.id}', event)" class="text-blue-400 hover:text-blue-300 font-medium">${DOMPurify.sanitize(link.name)}</a></td>
                        <td><span class="px-2 py-1 rounded text-xs text-white ${platformDetails.colorClass}"><i class="${platformDetails.iconClass} mr-1"></i>${platformDetails.name}</span></td>
                        <td class="text-center font-semibold">${link.visits}</td>
                        <td class="text-sm text-gray-400">${link.lastClick ? new Date(link.lastClick).toLocaleDateString() : 'Never'}</td>
                        <td>
                            <button onclick="editSocialLink('${link.id}')" class="btn btn-primary btn-sm !p-1.5 !text-xs mr-1"><i class="fas fa-edit"></i></button>
                            <button onclick="removeSocialLink('${link.id}')" class="btn btn-danger btn-sm !p-1.5 !text-xs"><i class="fas fa-trash"></i></button>
                        </td>
                    </tr>
                `;
            }).join('');
        }

        function trackSocialLinkClick(linkId, event) {
            const link = AppState.socialLinks.find(l => l.id === linkId);
            if (link) {
                link.visits++;
                link.lastClick = new Date().toISOString();
                saveToStorage('socialLinks', AppState.socialLinks);
                setTimeout(() => {
                    renderSocialLinksTable();
                    updateDashboardStats();
                }, 100);
            }
        }

        function editSocialLink(linkId) {
            const link = AppState.socialLinks.find(l => l.id === linkId);
            if (!link) return;
            document.getElementById('edit-link-id').value = link.id;
            document.getElementById('edit-link-name').value = link.name;
            document.getElementById('edit-link-url').value = link.url;
            document.getElementById('edit-link-platform').value = link.platform;
            document.getElementById('link-modal').style.display = 'block';
        }

        function saveSocialLinkEdit() {
            const id = document.getElementById('edit-link-id').value;
            const name = document.getElementById('edit-link-name').value.trim();
            const url = document.getElementById('edit-link-url').value.trim();
            const platform = document.getElementById('edit-link-platform').value;

            if (!name || !url || !isValidUrl(url)) {
                showAppNotification('Please enter a valid name and URL.', 'error');
                return;
            }

            const linkIndex = AppState.socialLinks.findIndex(l => l.id === id);
            if (linkIndex > -1) {
                AppState.socialLinks[linkIndex] = { ...AppState.socialLinks[linkIndex], name, url, platform };
                AppState.socialLinks.sort((a,b) => a.name.localeCompare(b.name));
                saveToStorage('socialLinks', AppState.socialLinks);
                renderSocialLinksTable();
                closeLinkModal();
                showAppNotification('Social link updated successfully!', 'success');
            }
        }

        function importAllData(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const importData = JSON.parse(e.target.result);
                    if (importData.appState) { 
                        Object.keys(AppState).forEach(key => {
                            if (importData.appState.hasOwnProperty(key)) {
                                if (typeof AppState[key] === 'object' && AppState[key] !== null && !Array.isArray(AppState[key]) &&
                                    typeof importData.appState[key] === 'object' && importData.appState[key] !== null && !Array.isArray(importData.appState[key])) {
                                    AppState[key] = { ...AppState[key], ...importData.appState[key] };
                                } else {
                                    AppState[key] = importData.appState[key];
                                }
                            }
                        });
                        AppState.settings = AppState.settings || { ...DEFAULT_SETTINGS };
                        AppState.trafficStats.chartData = (AppState.trafficStats.chartData && AppState.trafficStats.chartData.length === 7) ? AppState.trafficStats.chartData : [...DEFAULT_TRAFFIC_STATS.chartData];
                        AppState.newContentGenerator = AppState.newContentGenerator || JSON.parse(JSON.stringify(DEFAULT_NEW_CONTENT_GENERATOR_STATE));
                        AppState.newContentGenerator.selectedContentType = AppState.newContentGenerator.selectedContentType || DEFAULT_NEW_CONTENT_GENERATOR_STATE.selectedContentType;
                        AppState.adobeStockFiles = AppState.adobeStockFiles || [];
                        AppState.stats = AppState.stats || { ...DEFAULT_STATS };
                        AppState.generatedAIContent = AppState.generatedAIContent || [];


                        saveAllDataToStorage(); 
                        loadAndReinitializeUI(); 
                        showAppNotification('Data imported successfully! Application re-initialized.', 'success');
                    } else {
                        showAppNotification('Invalid data file format.', 'error');
                    }
                } catch (error) {
                    console.error('Import error:', error);
                    showAppNotification('Error importing data. File might be corrupted or invalid. ' + error.message, 'error');
                }
            };
            reader.readAsText(file);
            event.target.value = ''; 
        }
        
        let clearDataConfirmedOnce = false;
        function confirmAndClearAllData() {
            if(clearDataConfirmedOnce) return; 
            clearDataConfirmedOnce = true;

            if (confirm('DANGER! Are you absolutely sure you want to clear ALL application data? This action CANNOT be undone and will reset everything to defaults.')) {
                setTimeout(() => { 
                    if (confirm('FINAL WARNING: This will erase all your settings, tracked items, and generated content history. Proceed?')) {
                        clearAllApplicationData();
                    }
                    clearDataConfirmedOnce = false; 
                }, 50);
            } else {
                clearDataConfirmedOnce = false; 
            }
        }
        
        function clearAllApplicationData() {
            Object.keys(localStorage).forEach(key => {
                if (key.startsWith(STORAGE_PREFIX)) localStorage.removeItem(key);
            });
        
            AppState.trafficStats = { ...DEFAULT_TRAFFIC_STATS };
            AppState.socialLinks = [];
            AppState.adobeStockFiles = [];
            AppState.processedFiles = [];
            AppState.generatedAIContent = [];
            AppState.settings = { ...DEFAULT_SETTINGS };
            AppState.newContentGenerator = JSON.parse(JSON.stringify(DEFAULT_NEW_CONTENT_GENERATOR_STATE));
            AppState.stats = { ...DEFAULT_STATS };
            
            clearDataConfirmedOnce = false; 
            loadAndReinitializeUI(); 
            showAppNotification('All application data has been cleared!', 'warning');
        }

        function updateSystemInfo() {
            document.getElementById('app-version').textContent = APP_VERSION;
            document.getElementById('last-update-info').textContent = new Date().toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' });
            try {
                 document.getElementById('browser-info').textContent = navigator.userAgentData?.brands?.map(b => `${b.brand} ${b.version}`).join(', ') || navigator.userAgent.substring(0,50) + "...";
            } catch(e) { document.getElementById('browser-info').textContent = "N/A";}

            let totalStorageBytes = 0;
            for (let key in localStorage) {
                if (localStorage.hasOwnProperty(key) && key.startsWith(STORAGE_PREFIX)) {
                    totalStorageBytes += (localStorage[key].length * 2); 
                }
            }
            document.getElementById('storage-usage-info').textContent = formatFileSize(totalStorageBytes);
            document.getElementById('current-year').textContent = new Date().getFullYear();
        }
