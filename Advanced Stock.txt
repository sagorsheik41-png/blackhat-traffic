        async function processIndividualFile(file, options) {
            const extension = file.name.split('.').pop().toLowerCase();
            const baseName = file.name.replace(/\.[^/.]+$/, "");
            
            // Clean and optimize filename
            let cleanTitle = baseName
                .replace(/[-_]/g, ' ')
                .replace(/^\d+[\s\-_]*/, '')
                .trim()
                .replace(/\s+/g, ' ');

            // Generate SEO title
            let seoTitle = cleanTitle;
            if (options.optimizeTitles) {
                seoTitle = generateSEOTitle(cleanTitle, extension);
            }

            // Generate keywords
            let keywords = [];
            if (options.generateKeywords) {
                keywords = generateKeywords(seoTitle, extension);
            }

            // Generate description
            let description = '';
            if (options.generateDescriptions) {
                description = generateDescription(seoTitle, extension);
            }

            // Suggest categories
            let categories = [];
            if (options.suggestCategories) {
                categories = suggestCategories(seoTitle);
            }

            // Generate new filename - THIS IS THE KEY FIX
            let newFileName = file.name;
            if (options.renameFiles) {
                const seoFileName = seoTitle
                    .toLowerCase()
                    .replace(/[^\w\s-]/g, '') // Remove special characters except spaces and hyphens
                    .replace(/\s+/g, '-') // Replace spaces with hyphens
                    .replace(/-+/g, '-') // Replace multiple hyphens with single hyphen
                    .replace(/^-|-$/g, '') // Remove leading/trailing hyphens
                    .substring(0, 100); // Limit length for filesystem compatibility
                
                newFileName = `${seoFileName}.${extension}`;
            }

            // Read file content
            const fileContent = await readFileAsDataURL(file);

            // Process SVG metadata if needed
            let processedContent = fileContent;
            if (extension === 'svg' && options.embedMetadata) {
                processedContent = await embedSVGMetadata(fileContent, seoTitle, description);
            }

            return {
                originalFile: file,
                originalFileName: file.name,
                newFileName: newFileName, // This will be used for downloads
                title: seoTitle,
                description: description,
                keywords: keywords,
                categories: categories,
                fileType: extension,
                fileSize: file.size,
                content: processedContent,
                downloadUrl: null // Will be set when needed
            };
        }

        function generateSEOTitle(title, fileType) {

            let keywords = [...baseKeywords];

            const industry = detectIndustry(title);

            // Remove duplicates and limit to 30 keywords
            return [...new Set(keywords)].slice(0, 30);

        async function embedSVGMetadata(dataURL, title, description) {
            // For SVG files, we would embed title and description tags
            if (dataURL.startsWith('data:image/svg+xml')) {
                try {
                    const svgContent = atob(dataURL.split(',')[1]);
                    const parser = new DOMParser();
                    const svgDoc = parser.parseFromString(svgContent, 'image/svg+xml');
                    const svgElement = svgDoc.documentElement;
                    
                    // Add title element
                    let titleElement = svgDoc.createElement('title');
                    titleElement.textContent = title;
                    svgElement.insertBefore(titleElement, svgElement.firstChild);
                    
                    // Add description element
                    let descElement = svgDoc.createElement('desc');
                    descElement.textContent = description;
                    svgElement.insertBefore(descElement, titleElement.nextSibling);
                    
                    const serializer = new XMLSerializer();
                    const newSvgContent = serializer.serializeToString(svgDoc);
                    return 'data:image/svg+xml;base64,' + btoa(newSvgContent);
                } catch (error) {
                    console.error('Error embedding SVG metadata:', error);
                }
            }
            return dataURL;
        }

        async function downloadAllFiles() {

                showToast('No files to download', 'error');
                return;
            }

            showToast('Creating ZIP file with SEO-optimized names...', 'info');
            updateOverallStatus('Creating ZIP download...');
            
            try {
                const zip = new JSZip();
                
                // Add all processed files to ZIP with their SEO names
                for (const fileData of processedFiles) {
                    // Convert data URL to blob
                    const response = await fetch(fileData.content);
                    const blob = await response.blob();
                    
                    // Use the SEO-optimized filename in the ZIP
                    zip.file(fileData.newFileName, blob);
                }
                
                // Add metadata CSV
                const csvContent = generateCSVContent();
                zip.file('metadata.csv', csvContent);
                
                // Add metadata JSON
                const jsonContent = JSON.stringify(csvData, null, 2);
                zip.file('metadata.json', jsonContent);
                
                // Add filename mapping file
                const mappingContent = processedFiles.map(file => 
                    `${file.originalFileName} → ${file.newFileName}`
                ).join('\n');
                zip.file('filename-mapping.txt', mappingContent);
                
                // Generate and download ZIP
                const zipBlob = await zip.generateAsync({type: 'blob'});
                const link = document.createElement('a');
                link.href = URL.createObjectURL(zipBlob);
                link.download = 'seo-optimized-assets.zip';
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                
                // Update download counter
                downloadCount += processedFiles.length;
                document.getElementById('totalDownloads').textContent = downloadCount;
                
                updateOverallStatus('Download complete');
                showToast('All files downloaded with SEO-optimized names!', 'success');
                
            } catch (error) {
                console.error('Error creating ZIP:', error);
                showToast('Error creating ZIP file', 'error');
                updateOverallStatus('Download failed');
            }
        }

        function updateDownloadPreview() {
            const previewList = document.getElementById('downloadPreviewList');
            
            if (processedFiles.length === 0) {
                previewList.innerHTML = `
                    <div class="text-center py-12">
                        <i class="fas fa-download text-6xl text-gray-300 mb-4"></i>
                        <h3 class="text-xl font-semibold text-gray-600 mb-2">No downloads ready</h3>
                        <p class="text-gray-500">Process files to see download preview</p>
                    </div>
                `;
                return;
            }
            
            previewList.innerHTML = '';
            
            processedFiles.forEach((file, index) => {
                const previewItem = document.createElement('div');
                previewItem.className = 'download-preview';
                
                const fileTypeInfo = fileTypes[file.fileType] || fileTypes.default;
                
                previewItem.innerHTML = `
                    <div class="flex justify-between items-center mb-2">
                        <div class="flex items-center">
                            <span class="badge ${fileTypeInfo.badge} mr-3">${fileTypeInfo.name}</span>
                            <span class="font-semibold">${file.title}</span>
                        </div>
                        <button class="btn btn-success btn-sm" onclick="downloadSingleFile(${index})">
                            <i class="fas fa-download mr-1"></i>
                            Download
                        </button>
                    </div>
                    <div class="filename-change">
                        <span class="text-sm">Original:</span>
                        <span class="old-name">${file.originalFileName}</span>
                        <span class="arrow">→</span>
                        <span class="text-sm">SEO Download:</span>
                        <span class="new-name">${file.newFileName}</span>
                    </div>
                    <div class="text-xs text-gray-500 mt-2">
                        Size: ${Math.round(file.fileSize / 1024)} KB | 
                        Keywords: ${file.keywords.length} | 
                        Categories: ${file.categories.length}
                    </div>
                `;
                
                previewList.appendChild(previewItem);
            });
        }

        function downloadCsv() {
            const csvContent = generateCSVContent();
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = 'seo-metadata.csv';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            
            showToast('CSV downloaded successfully!', 'success');
        }

        function downloadJson() {
            const jsonContent = JSON.stringify(csvData, null, 2);
            const blob = new Blob([jsonContent], { type: 'application/json' });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = 'seo-metadata.json';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            
            showToast('JSON downloaded successfully!', 'success');
        }

        function generateCSVContent() {
            const headers = ['Original Name', 'SEO Optimized Name', 'Title', 'Description', 'Keywords', 'Categories', 'File Size', 'File Type'];
            const rows = csvData.map(row => [
                `"${row.originalName}"`,
                `"${row.newName}"`,
                `"${row.title}"`,
                `"${row.description}"`,
                `"${row.keywords}"`,
                `"${row.categories}"`,
                `"${row.fileSize}"`,
                `"${row.fileType}"`
            ]);
            
            return [headers.join(','), ...rows.map(row => row.join(','))].join('\n');
        }

        function updateStats() {
            const totalKeywords = processedFiles.reduce((sum, file) => sum + file.keywords.length, 0);
            document.getElementById('totalKeywords').textContent = totalKeywords;
            
            // Update analytics if visible
            if (document.getElementById('analyticsTab').classList.contains('active')) {
                updateAnalytics();
            }
            
            // Update previews if visible
            if (document.getElementById('exportTab').classList.contains('active')) {
                updatePreviews();
            }
        }

        function enableDownloadButtons() {
            document.getElementById('downloadAllBtn').disabled = false;
            document.getElementById('downloadCsvBtn').disabled = false;
            document.getElementById('downloadJsonBtn').disabled = false;
        }

        function startTimer() {
            timerInterval = setInterval(() => {
                if (startTime) {
                    const elapsed = Math.floor((Date.now() - startTime) / 1000);
                    const minutes = Math.floor(elapsed / 60).toString().padStart(2, '0');
                    const seconds = (elapsed % 60).toString().padStart(2, '0');
                    document.getElementById('processingTime').textContent = `${minutes}:${seconds}`;
                }
            }, 1000);
        }

        function stopTimer() {
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
        }

        function switchTab(tabName) {
            // Update tab buttons
            document.querySelectorAll('.tab-button').forEach(btn => {
                btn.classList.toggle('active', btn.dataset.tab === tabName);
            });
            
            // Update tab contents
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.toggle('active', content.id === tabName + 'Tab');
            });
            
            // Update specific content
            if (tabName === 'analytics') {
                updateAnalytics();
            } else if (tabName === 'export') {
                updatePreviews();
            } else if (tabName === 'downloads') {
                updateDownloadPreview();
            }
        }

        function updateAnalytics() {
            if (processedFiles.length === 0) return;
            
            // File type distribution
            const fileTypeCounts = {};
            processedFiles.forEach(file => {
                fileTypeCounts[file.fileType] = (fileTypeCounts[file.fileType] || 0) + 1;
            });
            
            updateFileTypeChart(fileTypeCounts);
            updateMetadataChart();
            updateTopKeywords();
        }

        function updateFileTypeChart(data) {
            const ctx = document.getElementById('fileTypeChart').getContext('2d');
            
            if (fileTypeChart) {
                fileTypeChart.destroy();
            }
            
            fileTypeChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: Object.keys(data).map(type => type.toUpperCase()),
                    datasets: [{
                        data: Object.values(data),
                        backgroundColor: Object.keys(data).map(type => fileTypes[type]?.color || '#6b7280'),
                        borderWidth: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            position: 'bottom'
                        }
                    }
                }
            });
        }

        function updateMetadataChart() {
            const ctx = document.getElementById('metadataChart').getContext('2d');
            
            if (metadataChart) {
                metadataChart.destroy();
            }
            
            const data = {
                'Titles': processedFiles.filter(f => f.title).length,
                'Descriptions': processedFiles.filter(f => f.description).length,
                'Keywords': processedFiles.filter(f => f.keywords.length > 0).length,
                'Categories': processedFiles.filter(f => f.categories.length > 0).length,
                'SEO Names': processedFiles.filter(f => f.newFileName !== f.originalFileName).length
            };
            
            metadataChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: Object.keys(data),
                    datasets: [{
                        label: 'Coverage',
                        data: Object.values(data),
                        backgroundColor: ['#3b82f6', '#10b981', '#f59e0b', '#8b5cf6', '#06b6d4'],
                        borderRadius: 8
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: {
                            beginAtZero: true,
                            max: processedFiles.length
                        }
                    },
                    plugins: {
                        legend: {
                            display: false
                        }
                    }
                }
            });
        }

        function updateTopKeywords() {
            const keywordCounts = {};
            processedFiles.forEach(file => {
                file.keywords.forEach(keyword => {
                    keywordCounts[keyword] = (keywordCounts[keyword] || 0) + 1;
                });
            });
            
            const topKeywords = Object.entries(keywordCounts)
                .sort((a, b) => b[1] - a[1])
                .slice(0, 20);
            
            const container = document.getElementById('topKeywords');
            container.innerHTML = topKeywords
                .map(([keyword, count]) => `<span class="keyword-tag">${keyword} (${count})</span>`)
                .join('');
        }

        function updatePreviews() {
            // CSV Preview
            const csvPreview = document.getElementById('csvPreview');
            const csvContent = generateCSVContent();
            const lines = csvContent.split('\n');
            csvPreview.textContent = lines.slice(0, 6).join('\n') + 
                (lines.length > 6 ? `\n... and ${lines.length - 6} more rows` : '');
            
            // JSON Preview
            const jsonPreview = document.getElementById('jsonPreview');
            const previewData = csvData.slice(0, 3);
            jsonPreview.textContent = JSON.stringify(previewData, null, 2);
        }

        function startNewBatch() {
            clearFiles();
            document.getElementById('successMessage').classList.add('hidden');
            downloadCount = 0;
            document.getElementById('totalDownloads').textContent = '0';
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function showToast(message, type = 'info') {
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            
            const icons = {
                success: 'check',
                error: 'times',
                warning: 'exclamation-triangle',
                info: 'info-circle'
            };
            
            toast.innerHTML = `
                <i class="fas fa-${icons[type]} mr-2"></i>
                ${message}
            `;
            
            document.body.appendChild(toast);
            
            setTimeout(() => {
                toast.style.opacity = '0';
                setTimeout(() => {
                    if (toast.parentNode) {
                        toast.remove();
                    }
                }, 300);
            }, 5000);
        }

        window.downloadSingleFile = downloadSingleFile;
