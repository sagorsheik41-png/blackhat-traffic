Don't forget to LIKE, SHARE, and SUBSCRIBE for more terrifying content every week!\n\n#DarkMysteriesUSA #${category.name.replace(' ', '')} #${keywords[0].replace(' ', '')}`,
        
        `${currentTitle} - This ${category.name.toLowerCase()} investigation will change everything you thought you knew.\n\nWe've uncovered new evidence that sheds light on this decades-old mystery. What really happened, and why has the truth been hidden for so long?\n\nJoin us as we piece together the clues and reveal what they don't want you to know.\n\nIf you enjoy our content, please consider subscribing for more mysteries every week!\n\n#DarkMysteriesUSA #${category.name.replace(' ', '')} #${keywords[0].replace(' ', '')}`,

This case has puzzled investigators for years, but we may have found the missing piece of the puzzle.\n\n${getQuestion(keywords)}\n${getQuestion(keywords)}\n\nDon't forget to hit the LIKE button and SUBSCRIBE for more content that will keep you up at night!\n\n#DarkMysteriesUSA #${category.name.replace(' ', '')} #${keywords[0].replace(' ', '')}`
      ];
      
      currentDescription = descriptionTemplates[Math.floor(Math.random() * descriptionTemplates.length)];
      descContent.textContent = currentDescription;
      updateSEOScores();

    }

    // Generate keywords based on current content
    function generateKeywords() {
      if (!currentTitle) {
        generateTitle();
      }
      
      const category = videoCategories[currentCategory];
      const keywords = extractKeywords(currentHistory);
      
      const baseKeywords = [
        "DarkMysteriesUSA", 
        category.name.replace(' ', ''), 
        ...keywords.slice(0, 3),
        "Mystery", "Investigation", "Unsolved", "TrueCrime", "Documentary"
      ];
      
      const additionalKeywords = [
        "Paranormal", "Supernatural", "Conspiracy", "Haunting", "Ghost",
        "ColdCase", "SerialKiller", "Police", "Detective", "Evidence",
        "Shocking", "Revelation", "Secret", "CoverUp", "HiddenTruth"
      ];
      
      // Select 5-8 additional keywords
      const shuffled = [...additionalKeywords].sort(() => 0.5 - Math.random());
      const selected = shuffled.slice(0, 5 + Math.floor(Math.random() * 4));
      
      currentKeywords = [...baseKeywords, ...selected].join(', ');
      keywordsContent.textContent = currentKeywords;
      updateSEOScores();
      updateStepIndicators();
    }

    // Update all content displays
    function updateAllContent() {
      titleContent.textContent = currentTitle || "Title will be generated here...";
      descContent.textContent = currentDescription || "Description will be generated here...";
      keywordsContent.textContent = currentKeywords || "Keywords will be generated here...";
      updateSEOScores();
      updateStepIndicators();
      updateThumbnailText();
    }

    // Copy content to clipboard
    function copyContent(type) {
      let content = '';
      
      switch(type) {
        case 'history':
          content = currentHistory;
          break;
        case 'title':
          content = currentTitle;
          break;
        case 'description':
          content = currentDescription;
          break;
        case 'keywords':
          content = currentKeywords;
          break;
      }
      
      if (!content) {
        showNotification(`Please generate ${type} first!`, 'warning');
        return;
      }
      
      // Create a temporary textarea to copy from
      const textarea = document.createElement('textarea');
      textarea.value = content;
      document.body.appendChild(textarea);
      textarea.select();
      
      try {
        const successful = document.execCommand('copy');
        if (successful) {
          showNotification(`${type.charAt(0).toUpperCase() + type.slice(1)} copied to clipboard!`, 'success');
        }
      } catch (err) {
        console.error('Failed to copy text: ', err);
        showNotification('Failed to copy text. Please try again.', 'warning');
      }
      
      document.body.removeChild(textarea);
    }

    // Update SEO scores
    function updateSEOScores() {
      // Calculate title score
      const titleLengthValue = currentTitle ? currentTitle.length : 0;
      titleLength.textContent = titleLengthValue;
      
      let titleScoreValue = 0;
      if (titleLengthValue > 0) {
        if (titleLengthValue <= 70) titleScoreValue = 100;
        else if (titleLengthValue <= 80) titleScoreValue = 80;
        else titleScoreValue = 60;
      }
      
      // Update title score display
      titleScore.textContent = `SEO: ${getScoreText(titleScoreValue)}`;
      titleScore.className = `score-${getScoreLevel(titleScoreValue)}`;
      
      // Calculate description score
      const descLengthValue = currentDescription ? currentDescription.length : 0;
      descLength.textContent = descLengthValue;
      
      let descScoreValue = 0;
      if (descLengthValue > 0) {
        if (descLengthValue >= 150 && descLengthValue <= 5000) descScoreValue = 100;
        else if (descLengthValue > 0) descScoreValue = 70;
      }
      
      // Update description score display
      descScore.textContent = `SEO: ${getScoreText(descScoreValue)}`;
      descScore.className = `score-${getScoreLevel(descScoreValue)}`;
      
      // Calculate keywords score
      const keywordsValue = currentKeywords ? currentKeywords.split(',').length : 0;
      keywordsCount.textContent = keywordsValue;
      
      let keywordsScoreValue = 0;
      if (keywordsValue > 0) {
        if (keywordsValue >= 5 && keywordsValue <= 15) keywordsScoreValue = 100;
        else if (keywordsValue > 0) keywordsScoreValue = 70;
      }
      
      // Update keywords score display
      keywordsScore.textContent = `SEO: ${getScoreText(keywordsScoreValue)}`;
      keywordsScore.className = `score-${getScoreLevel(keywordsScoreValue)}`;
      
      // Calculate overall SEO score
      const overallScore = Math.round((titleScoreValue + descScoreValue + keywordsScoreValue) / 3);
      seoScore.textContent = overallScore;
      
      // Calculate viral potential
      const viralPotential = Math.min(100, overallScore + 20);
      viralScore.textContent = viralPotential;
    }

    // Update step indicators
    function updateStepIndicators() {
      steps.forEach(step => {
        const stepNum = parseInt(step.dataset.step);
        step.classList.remove('active', 'completed');
        
        if (stepNum === 1) {
          if (currentTitle) step.classList.add('completed');
          else step.classList.add('active');
        } else if (stepNum === 2) {
          if (currentDescription) step.classList.add('completed');
          else if (currentTitle) step.classList.add('active');
        } else if (stepNum === 3) {
          if (currentKeywords) step.classList.add('completed');
          else if (currentDescription) step.classList.add('active');
        }
      });
    }

    // Update thumbnail text
    function updateThumbnailText() {
      if (!currentTitle) return;
      
      const keywords = extractKeywords(currentTitle);
      thumbnailText.textContent = `Thumbnail suggestion: Use a dark, mysterious background with bold text "${keywords[0]}" and an intriguing image related to ${videoCategories[currentCategory].name.toLowerCase()}. Add a red arrow or circle highlighting a key element.`;
    }

    // Helper functions
    function extractKeywords(text) {
      const words = text.toLowerCase().match(/\b(\w+)\b/g) || [];
      const filtered = words.filter(word => word.length > 4);
      return [...new Set(filtered)].slice(0, 5);
    }

    function getQuestion(keywords) {
      const questions = [
        `What really happened with the ${keywords[0]}?`,
        `Why did the ${keywords[1]} remain unsolved?`,
        `Is there a connection between the ${keywords[0]} and ${keywords[2]}?`,
        `What evidence points to a ${keywords[3]}?`,
        `Could ${keywords[4]} be the key to solving this mystery?`,
        `What don't they want you to know about ${keywords[0]}?`
      ];
      return questions[Math.floor(Math.random() * questions.length)];
    }

    function getScoreText(score) {
      if (score >= 90) return "Excellent";
      if (score >= 70) return "Good";
      if (score >= 50) return "Fair";
      return "Poor";
    }

    function getScoreLevel(score) {
      if (score >= 80) return "high";
      if (score >= 60) return "medium";
      return "low";
    }

    function showNotification(message, type = 'success') {
      notification.textContent = message;
      notification.className = 'notification';
      notification.classList.add(type);
      
      // Change icon based on type
      const icon = notification.querySelector('i');
      icon.className = type === 'success' ? 'fas fa-check-circle' : 
                       type === 'warning' ? 'fas fa-exclamation-triangle' : 'fas fa-exclamation-circle';
      
      notification.classList.add('show');
      
      setTimeout(() => {
        notification.classList.remove('show');
      }, 3000);
    }

    // Initialize the application when the DOM is loaded
    document.addEventListener('DOMContentLoaded', init);
