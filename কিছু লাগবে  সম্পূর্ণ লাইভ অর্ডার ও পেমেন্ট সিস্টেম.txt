        function addProduct() {
            const name = document.getElementById('productName').value.trim();
            const qty = parseInt(document.getElementById('productQty').value) || 1;
            const price = parseFloat(document.getElementById('productPrice').value) || 0;
            if (!name) {

                return;

            if (price <= 0) {

            const product = {

        }

        function calculateOrder() {
            const customerName = document.getElementById('customerName').value.trim();
            const customerPhone = document.getElementById('customerPhone').value.trim();
            const customerAddress = document.getElementById('customerAddress').value.trim();
            if (!customerName || !customerPhone || !customerAddress) {
                showNotification('গ্রাহকের সকল তথ্য পূরণ করুন', 'error');
                return;
            }
            if (products.length === 0) {
                showNotification('কমপক্ষে একটি পণ্য যোগ করুন', 'error');
                return;
            }
            const subtotal = products.reduce((sum, product) => sum + product.total, 0);
            const deliveryCharge = parseFloat(document.getElementById('deliveryCharge').value) || 0;
            const serviceChargePercent = parseFloat(document.getElementById('serviceCharge').value) || 0;
            const serviceChargeAmount = (subtotal * serviceChargePercent) / 100;
            const grandTotal = subtotal + deliveryCharge + serviceChargeAmount;
            currentOrder = {
                id: generateOrderId(),
                timestamp: new Date().toISOString(),
                customer: {
                    name: customerName,
                    phone: customerPhone,
                    address: customerAddress
                },
                products: [...products],
                charges: {
                    subtotal,
                    deliveryCharge,
                    serviceChargePercent,
                    serviceChargeAmount,
                    grandTotal
                },
                status: 'Pending',
                paymentStatus: 'Unpaid'
            };
            showInvoicePreview();
            document.getElementById('proceedPaymentBtn').disabled = false;
            showNotification('অর্ডার প্রস্তুত! এখন পেমেন্ট করুন');
        }

        }

        function selectPaymentMethod(method) {
            document.querySelectorAll('.payment-method').forEach(el => el.classList.remove('selected'));
            event.currentTarget.classList.add('selected');
            const methodName = method === 'bkash' ? 'বিকাশ' : 'নগদ';
            const number = CONFIG.BUSINESS[method];
            document.getElementById('paymentInstructions').innerHTML = `
                ${methodName} সেন্ড মানি করুন: <strong>${number}</strong><br>
                পরিমাণ: <strong>${formatCurrency(currentOrder.charges.grandTotal)}</strong>
            `;
            document.getElementById('paymentForm').style.display = 'block';
        }

        function confirmPayment() {
            const transactionId = document.getElementById('transactionId').value.trim();
            const senderNumber = document.getElementById('senderNumber').value.trim();
            if (!transactionId || !senderNumber) {
                showNotification('সব তথ্য পূরণ করুন', 'error');
                return;
            }
            currentOrder.paymentStatus = 'Paid';
            currentOrder.payment = {
                transactionId,
                senderNumber,
                method: document.querySelector('.payment-method.selected') ? 
                       document.querySelector('.payment-method.selected').textContent.includes('বিকাশ') ? 'bkash' : 'nagad' : 'unknown',
                timestamp: new Date().toISOString()
            };
            saveOrderToSheet(currentOrder);
            closePaymentModal();
            resetOrderForm();
            showNotification('অর্ডার সফলভাবে সম্পন্ন হয়েছে! আমরা শীঘ্রই যোগাযোগ করবো।', 'success');
        }

        function closePaymentModal() {
            document.getElementById('paymentModal').classList.remove('active');
            document.getElementById('paymentForm').style.display = 'none';
            document.querySelectorAll('.payment-method').forEach(el => el.classList.remove('selected'));
            document.getElementById('transactionId').value = '';
            document.getElementById('senderNumber').value = '';
        }

        function resetOrderForm() {
            products = [];
            currentOrder = null;
            document.getElementById('customerName').value = '';
            document.getElementById('customerPhone').value = '';
            document.getElementById('customerAddress').value = '';
            document.getElementById('deliveryCharge').value = '60';
            document.getElementById('serviceCharge').value = '12';
            document.getElementById('proceedPaymentBtn').disabled = true;

                </div>
            `;
        }

        // ==================== GOOGLE SHEETS INTEGRATION ====================
        async function initializeGoogleAPI() {
            try {
                await gapi.load('client', async () => {
                    await gapi.client.init({
                        apiKey: CONFIG.API_KEY,
                        discoveryDocs: ['https://sheets.googleapis.com/$discovery/rest?version=v4']
                    });
                });
                tokenClient = google.accounts.oauth2.initTokenClient({
                    client_id: CONFIG.CLIENT_ID,
                    scope: 'https://www.googleapis.com/auth/spreadsheets',
                    callback: (tokenResponse) => {
                        if (tokenResponse.error) {
                            showNotification('Google সাইন ইন ব্যর্থ হয়েছে', 'error');
                            return;
                        }
                        accessToken = tokenResponse.access_token;
                        // টোকেন স্টোর করুন (একবার লগইন করলে আর লগইন করতে হবে না)
                        localStorage.setItem('gapi_access_token', accessToken);
                        showNotification('Google Sheets এর সাথে সংযুক্ত হয়েছে', 'success');
                        if (document.getElementById('adminPanel').classList.contains('active')) {
                            loadOrders();
                        }
                    }
                });
                google.accounts.id.initialize({
                    client_id: CONFIG.CLIENT_ID,
                    callback: () => tokenClient.requestAccessToken()
                });
                google.accounts.id.renderButton(document.getElementById('googleSignIn'), {
                    theme: 'outline',
                    size: 'medium',
                    text: 'signin_with',
                    logo_alignment: 'left'
                });

                // পূর্বের টোকেন আছে কিনা চেক করুন
                const savedToken = localStorage.getItem('gapi_access_token');
                if (savedToken) {
                    accessToken = savedToken;
                    showNotification('স্বয়ংক্রিয়ভাবে Google সাইন ইন হয়েছে', 'success');
                    if (document.getElementById('adminPanel').classList.contains('active')) {
                        loadOrders();
                    }
                }
            } catch (error) {
                console.error('Google API initialization failed:', error);
                showNotification('Google API সংযোগে সমস্যা', 'error');
            }
        }

        async function saveOrderToSheet(order) {
            if (!accessToken) {
                showNotification('Google সাইন ইন প্রয়োজন', 'warning');
                tokenClient.requestAccessToken();
                return;
            }
            try {
                const values = [[
                    order.id,
                    order.timestamp,
                    order.customer.name,
                    order.customer.phone,
                    order.customer.address,
                    order.charges.subtotal,
                    order.charges.deliveryCharge,
                    order.charges.serviceChargeAmount,
                    order.charges.grandTotal,
                    order.status,
                    order.paymentStatus,
                    JSON.stringify(order.products),
                    order.payment ? order.payment.transactionId : '',
                    order.payment ? order.payment.senderNumber : '',
                    order.payment ? order.payment.method : ''
                ]];
                await gapi.client.sheets.spreadsheets.values.append({
                    spreadsheetId: CONFIG.SHEET_ID,
                    range: 'Orders!A:O',
                    valueInputOption: 'USER_ENTERED',
                    resource: { values }
                });
                orders.unshift(order);
                showNotification('অর্ডার Google Sheets এ সেভ হয়েছে', 'success');
            } catch (error) {
                console.error('Sheet save error:', error);
                showNotification('Google Sheets এ সেভ করতে সমস্যা হয়েছে', 'error');
            }
        }

        async function loadOrders() {
            if (!accessToken) return;
            try {
                const response = await gapi.client.sheets.spreadsheets.values.get({
                    spreadsheetId: CONFIG.SHEET_ID,
                    range: 'Orders!A:O'
                });
                const rows = response.result.values || [];
                if (rows.length <= 1) {
                    orders = [];
                } else {
                    orders = rows.slice(1).map(row => ({
                        id: row[0],
                        timestamp: row[1],
                        customer: {
                            name: row[2],
                            phone: row[3],
                            address: row[4]
                        },
                        charges: {
                            subtotal: parseFloat(row[5] || 0),
                            deliveryCharge: parseFloat(row[6] || 0),
                            serviceChargeAmount: parseFloat(row[7] || 0),
                            grandTotal: parseFloat(row[8] || 0)
                        },
                        status: row[9] || 'Pending',
                        paymentStatus: row[10] || 'Unpaid',
                        products: JSON.parse(row[11] || '[]'),
                        payment: row[12] ? {
                            transactionId: row[12],
                            senderNumber: row[13],
                            method: row[14]
                        } : null
                    })).reverse();
                }
                updateAdminStats();
                updateOrdersTable();
            } catch (error) {
                console.error('Load orders error:', error);
                showNotification('অর্ডার লোড করতে সমস্যা', 'error');
            }
        }

        // ==================== ADMIN PANEL ====================
        function toggleAdminPanel() {
            const customerInterface = document.getElementById('customerInterface');
            const adminPanel = document.getElementById('adminPanel');
            if (adminPanel.classList.contains('active')) {
                adminPanel.classList.remove('active');
                customerInterface.style.display = 'block';
                isAdminLoggedIn = false;
                showNotification('এডমিন প্যানেল থেকে বের হয়েছেন', 'warning');
            } else {
                if (isAdminLoggedIn) {
                    showAdminView();
                } else {
                    document.getElementById('adminLoginModal').classList.add('active');
                }
            }
        }

        function handleAdminLogin() {
            const username = document.getElementById('adminUsername').value;
            const password = document.getElementById('adminPassword').value;
            if (username === CONFIG.ADMIN.username && password === CONFIG.ADMIN.password) {
                isAdminLoggedIn = true;
                closeAdminLoginModal();
                showNotification('লগইন সফল হয়েছে!', 'success');
                showAdminView();
            } else {
                showNotification('ভুল ইউজারনেম বা পাসওয়ার্ড!', 'error');
                document.getElementById('adminPassword').value = '';
            }
        }

        function showAdminView() {
            document.getElementById('customerInterface').style.display = 'none';
            document.getElementById('adminPanel').classList.add('active');
            if (accessToken) {
                loadOrders();
            } else {
                showNotification('ডেটা দেখতে Google এ সাইন ইন করুন', 'warning');
            }
        }

        function closeAdminLoginModal() {
            document.getElementById('adminLoginModal').classList.remove('active');
            document.getElementById('adminUsername').value = '';
            document.getElementById('adminPassword').value = '';
        }

        function updateAdminStats() {
            const totalOrders = orders.length;
            const pendingOrders = orders.filter(o => o.status === 'Pending').length;
            const totalRevenue = orders.filter(o => o.paymentStatus === 'Paid').reduce((sum, o) => sum + o.charges.grandTotal, 0);
            const today = new Date().toDateString();
            const todayOrders = orders.filter(o => new Date(o.timestamp).toDateString() === today).length;
            document.getElementById('totalOrders').textContent = totalOrders;
            document.getElementById('pendingOrders').textContent = pendingOrders;
            document.getElementById('totalRevenue').innerHTML = formatCurrency(totalRevenue);
            document.getElementById('todayOrders').textContent = todayOrders;
        }

        function updateOrdersTable() {
            const tbody = document.getElementById('ordersTableBody');
            if (orders.length === 0) {
                tbody.innerHTML = `
                    <tr>
                        <td colspan="7" class="px-4 py-6 text-center text-gray-500">
                            <i class="fas fa-inbox text-3xl mb-2 block"></i>
                            কোন অর্ডার পাওয়া যায়নি
                        </td>
                    </tr>
                `;
                return;
            }
            tbody.innerHTML = orders.slice(0, 20).map(order => `
                <tr class="hover:bg-gray-50">
                    <td class="px-4 py-3 text-sm font-medium text-blue-600">${order.id}</td>
                    <td class="px-4 py-3 text-sm text-gray-900">${order.customer.name}</td>
                    <td class="px-4 py-3 text-sm text-gray-900">${order.customer.phone}</td>
                    <td class="px-4 py-3 text-sm font-medium text-gray-900 text-right">${formatCurrency(order.charges.grandTotal)}</td>
                    <td class="px-4 py-3 text-center">
                        <span class="px-2 py-1 text-xs font-medium rounded-full ${
                            order.status === 'Delivered' ? 'bg-green-100 text-green-800' :
                            order.status === 'Processing' ? 'bg-yellow-100 text-yellow-800' :
                            'bg-gray-100 text-gray-800'
                        }">${order.status}</span>
                    </td>
                    <td class="px-4 py-3 text-center">
                        <span class="px-2 py-1 text-xs font-medium rounded-full ${
                            order.paymentStatus === 'Paid' ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'
                        }">${order.paymentStatus}</span>
                    </td>
                    <td class="px-4 py-3 text-sm text-gray-500">${formatDate(order.timestamp)}</td>
                </tr>
            `).join('');
        }
