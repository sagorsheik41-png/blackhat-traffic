            const shuffled = [...array];
            for (let i = shuffled.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
            }
            return shuffled;
        }

        function copyToClipboard(text, button) {
            navigator.clipboard.writeText(text).then(() => {
                const originalHTML = button.innerHTML;
                button.innerHTML = '<i class="fas fa-check mr-2"></i>Copied!';
                button.style.transform = 'scale(1.05)';
                
                setTimeout(() => {
                    button.innerHTML = originalHTML;
                    button.style.transform = 'scale(1)';
                }, 2500);
                
                showNotification('Copied to clipboard successfully!', 'success');
            }).catch(err => {
                console.error('Copy failed:', err);
                showNotification('Failed to copy to clipboard', 'error');
            });
        }

        function initializeUploadTracker() {
            const addFileBtn = document.getElementById('add-file');
            const uploadDateInput = document.getElementById('upload-date');
            
            // Set today's date as default
            uploadDateInput.valueAsDate = new Date();
            
            addFileBtn.addEventListener('click', addFile);
            
            // Load saved files
            loadUploadedFiles();
            renderUploadedFiles();
        }

        function addFile() {
            const fileName = document.getElementById('file-name').value.trim();
            const uploadDate = document.getElementById('upload-date').value;
            const status = document.getElementById('file-status').value;
            const downloadCount = parseInt(document.getElementById('download-count').value) || 0;

            if (!fileName || !uploadDate) {
                showNotification('Please enter file name and upload date', 'error');
                return;
            }

            const newFile = {
                id: Date.now(),
                name: fileName,
                uploadDate: uploadDate,
                status: status,
                downloads: downloadCount,
                estimatedRevenue: calculateEstimatedRevenue(downloadCount, status),
                addedAt: new Date().toISOString()
            };

            AppState.uploadedFiles.push(newFile);
            saveUploadedFiles();
            
            // Clear form
            document.getElementById('file-name').value = '';
            document.getElementById('upload-date').valueAsDate = new Date();
            document.getElementById('download-count').value = '';
            
            renderUploadedFiles();
            updateStats();
            showNotification('Asset added to tracker successfully!', 'success');
        }

        function calculateEstimatedRevenue(downloads, status) {
            if (status !== 'Approved' && status !== 'Live') return 0;
            
            // Average revenue per download varies, using conservative estimate
            const avgRevenuePerDownload = 0.33; // $0.33 average
            return downloads * avgRevenuePerDownload;
        }

        function renderUploadedFiles() {
            const tbody = document.getElementById('file-tracker-body');
            
            if (AppState.uploadedFiles.length === 0) {
                tbody.innerHTML = `
                    <tr>
                        <td colspan="6" style="text-align: center; color: #94a3b8; padding: 50px; font-size: 1.2rem;">
                            <i class="fas fa-upload" style="font-size: 3rem; margin-bottom: 20px; display: block;"></i>
                            Start tracking your Adobe Stock uploads for better performance insights
                        </td>

                            ${file.status}

                    <td style="text-align: center; font-weight: 700; color: #3b82f6;">${file.downloads.toLocaleString()}</td>

        function getStatusStyle(status) {
            const styles = {
                'Uploaded': 'background: linear-gradient(135deg, #3b82f6, #1d4ed8); color: white;',
                'Under Review': 'background: linear-gradient(135deg, #f59e0b, #d97706); color: white;',
                'Approved': 'background: linear-gradient(135deg, #10b981, #059669); color: white;',
                'Rejected': 'background: linear-gradient(135deg, #ef4444, #dc2626); color: white;',
                'Live': 'background: linear-gradient(135deg, #8b5cf6, #7c3aed); color: white;'
            };
            return styles[status] || styles['Uploaded'];
        }

        function removeFile(id) {
            if (confirm('Are you sure you want to remove this file from tracking?')) {
                AppState.uploadedFiles = AppState.uploadedFiles.filter(file => file.id !== id);
                saveUploadedFiles();
                renderUploadedFiles();
                updateStats();
                showNotification('File removed from tracker', 'warning');
            }
        }

        function saveUploadedFiles() {
            localStorage.setItem('uploadedFiles', JSON.stringify(AppState.uploadedFiles));
        }

        function loadUploadedFiles() {
            const saved = localStorage.getItem('uploadedFiles');
            if (saved) {
                AppState.uploadedFiles = JSON.parse(saved);
            }
        }

        // ============================================= 
        // Stats Management
        // ============================================= 
        function updateStats() {
            const files = AppState.uploadedFiles;
            
            AppState.stats.totalAssets = files.length;
            AppState.stats.totalDownloads = files.reduce((sum, file) => sum + file.downloads, 0);
            AppState.stats.totalEarnings = files.reduce((sum, file) => sum + file.estimatedRevenue, 0);
            
            const approvedFiles = files.filter(file => file.status === 'Approved' || file.status === 'Live');
            AppState.stats.acceptanceRate = files.length > 0 ? (approvedFiles.length / files.length) * 100 : 0;
            
            // Update display
            document.getElementById('total-earnings').textContent = `$${AppState.stats.totalEarnings.toFixed(2)}`;
            document.getElementById('total-downloads').textContent = AppState.stats.totalDownloads.toLocaleString();
            document.getElementById('total-assets').textContent = AppState.stats.totalAssets.toLocaleString();
            document.getElementById('acceptance-rate').textContent = `${AppState.stats.acceptanceRate.toFixed(1)}%`;
        }

            const generateBtn = document.getElementById('generate-keywords');

        function generateKeywords() {
            const description = document.getElementById('content-description').value.trim();
            const category = document.getElementById('keyword-category').value;
            
            if (!description) {
                showNotification('Please describe your content concept first', 'error');
                return;
            }
            
            // Extract words from description
            const words = description.toLowerCase().match(/\b\w+\b/g) || [];
            const uniqueWords = [...new Set(words)].filter(word => word.length > 2);
            
            // Base keywords
            const baseKeywords = ['vector', 'illustration', 'design', 'graphic', 'template', 'professional', 'modern', 'clean', 'quality', 'premium'];
            
            // Category-specific keywords
            const categoryKeywords = getCategoryKeywords(category, '', '');
            
            // Style keywords
            const styleKeywords = ['minimalist', 'contemporary', 'elegant', 'sophisticated', 'creative', 'artistic', 'innovative', 'unique', 'stylish', 'trendy'];
            
            // Purpose keywords
            const purposeKeywords = ['commercial', 'business', 'corporate', 'marketing', 'branding', 'presentation', 'website', 'social media', 'print', 'digital'];
            
            // Technical keywords
            const technicalKeywords = ['high-resolution', 'scalable', 'editable', 'customizable', 'layered', 'RGB', 'CMYK', 'print-ready', 'web-ready'];
            
            // Combine all keywords
            const allKeywords = [
                ...baseKeywords,
                ...categoryKeywords,
                ...styleKeywords,
                ...purposeKeywords,
                ...technicalKeywords,
                ...uniqueWords.slice(0, 10) // Top 10 words from description
            ];
            
            // Remove duplicates and create final list
            const uniqueKeywords = [...new Set(allKeywords)];
            const finalKeywords = shuffleArray(uniqueKeywords).slice(0, 50);
            
            displayKeywords(finalKeywords);
            showNotification(`Generated ${finalKeywords.length} optimized keywords!`, 'success');
        }

        function displayKeywords(keywords) {
            const container = document.getElementById('keyword-container');
            const resultsDiv = document.getElementById('keyword-results');
            
            container.innerHTML = keywords.map(keyword => `<span class="keyword-tag">${keyword}</span>`).join('');
            resultsDiv.style.display = 'block';
            
            // Store keywords for copy/export
            window.currentKeywords = keywords;
        }

        function copyAllKeywords() {
            if (!window.currentKeywords) {
                showNotification('Generate keywords first', 'error');
                return;
            }
            
            const keywordText = window.currentKeywords.join(', ');
            copyToClipboard(keywordText, document.getElementById('copy-keywords'));
        }

        function exportKeywordsCSV() {
            if (!window.currentKeywords) {
                showNotification('Generate keywords first', 'error');
                return;
            }
            
            const csvContent = [
                ['Keyword', 'Category', 'Type'],
                ...window.currentKeywords.map(keyword => [keyword, 'Generated', 'Primary'])
            ].map(row => row.join(',')).join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `adobe-stock-keywords-${new Date().toISOString().split('T')[0]}.csv`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            showNotification('Keywords exported as CSV!', 'success');
        }
