        // Global variables
        let currentUserAgent = '';
        let currentMethod = 'direct';
        let autoReloadTimer = null;
        let startTime = null;
        let uptimeTimer = null;
        let stats = {
            loadCount: 0,
            successCount: 0,
            errorCount: 0
        };

        // Initialize
        document.addEventListener('DOMContentLoaded', function() {
            updateUserAgent();
            updateStats();
            loadSettings();
        });

        function selectMethod(method) {
            currentMethod = method;
            document.querySelectorAll('.method-option').forEach(option => {
                option.classList.remove('active');
            });
            document.querySelector(`[data-method="${method}"]`).classList.add('active');
            saveSettings();
        }

        function updateUserAgent() {
            const uaType = document.getElementById('uaType').value;
            let userAgent = '';

            if (uaType === 'custom') {
                userAgent = document.getElementById('customUserAgent').value.trim();
                if (!userAgent) {
                    userAgent = getRandomUserAgent('chrome-desktop');
                }
                document.getElementById('customUaSection').style.display = 'block';
            } else {
                document.getElementById('customUaSection').style.display = 'none';
                if (uaType === 'random') {
                    const types = Object.keys(userAgents);
                    const randomType = types[Math.floor(Math.random() * types.length)];
                    userAgent = getRandomUserAgent(randomType);
                } else {
                    userAgent = getRandomUserAgent(uaType);
                }
            }

            currentUserAgent = userAgent;
            document.getElementById('currentUserAgent').textContent = userAgent;
        }

        function getRandomUserAgent(type) {
            const agents = userAgents[type] || userAgents['chrome-desktop'];
            return agents[Math.floor(Math.random() * agents.length)];
        }

        function startPreview() {
            const url = document.getElementById('targetUrl').value.trim();
            
            if (!url) {
                showNotification('Please enter a URL | একটি URL প্রবেশ করান', 'error');
                return;
            }

            if (!isValidUrl(url)) {
                showNotification('Please enter a valid URL | একটি বৈধ URL প্রবেশ করান', 'error');
                return;
            }

            if (!startTime) {
                startTime = Date.now();
                startUptimeTimer();
            }

            loadPreview(url);
            
            if (document.getElementById('autoReload').checked) {
                setupAutoReload();
            }
        }

        function loadPreview(url) {
            stats.loadCount++;
            updateStats();

            const loadingOverlay = document.getElementById('loadingOverlay');
            const errorDisplay = document.getElementById('errorDisplay');
            const previewContent = document.getElementById('previewContent');

            loadingOverlay.style.display = 'flex';
            errorDisplay.style.display = 'none';
            document.getElementById('statusText').textContent = 'Loading... | লোড হচ্ছে...';
            document.getElementById('currentUrl').textContent = url;

            // Create preview content
            setTimeout(() => {
                try {
                    const previewUrl = getPreviewUrl(url);
                    const htmlContent = createPreviewContent(url, previewUrl);
                    
                    previewContent.innerHTML = htmlContent;
                    loadingOverlay.style.display = 'none';
                    document.getElementById('statusText').textContent = 'Loaded | লোড সম্পন্ন';
                    document.getElementById('lastUpdated').textContent = new Date().toLocaleTimeString();
                    
                    stats.successCount++;
                    updateStats();
                    showNotification('Preview loaded | প্রিভিউ লোড হয়েছে', 'success');
                } catch (error) {
                    loadingOverlay.style.display = 'none';
                    errorDisplay.style.display = 'flex';
                    document.getElementById('statusText').textContent = 'Error | ত্রুটি';
                    
                    stats.errorCount++;
                    updateStats();
                    showNotification('Failed to load | লোড ব্যর্থ', 'error');
                }
            }, 2000);
        }

        function getPreviewUrl(url) {
            if (currentMethod === 'proxy') {
                return `https://api.allorigins.win/get?url=${encodeURIComponent(url)}`;
            }
            return url;
        }

        function createPreviewContent(targetUrl, previewUrl) {
            const loadTime = new Date().toLocaleString();
            
            return `
                <div class="h-full bg-white">
                    <div class="p-4 bg-gradient-to-r from-blue-50 to-indigo-50 border-b">
                        <h3 class="text-lg font-bold text-blue-800 mb-2">
                            <i class="fas fa-info-circle mr-2"></i>Preview Information
                        </h3>
                        <div class="grid grid-cols-1 md:grid-cols-2 gap-4 text-sm">
                            <div>
                                <p><strong>Target URL:</strong></p>
                                <p class="text-blue-600 break-all font-mono text-xs">${targetUrl}</p>
                                <p class="mt-2"><strong>Method:</strong> ${currentMethod === 'direct' ? 'Direct iframe' : 'Proxy service'}</p>
                                <p><strong>Load Time:</strong> ${loadTime}</p>
                            </div>
                            <div>
                                <p><strong>User-Agent:</strong></p>
                                <p class="text-gray-600 break-all font-mono text-xs">${currentUserAgent}</p>
                            </div>
                        </div>
                    </div>
                    <div class="iframe-container">
                        ${createIframeContent(previewUrl, targetUrl)}
                    </div>
                </div>
            `;
        }

        function createIframeContent(previewUrl, targetUrl) {
            return `
                <iframe 
                    src="${previewUrl}" 
                    class="w-full h-full"
                    onload="handleIframeLoad(this)"
                    onerror="handleIframeError(this)"
                    sandbox="allow-scripts allow-same-origin allow-forms"
                ></iframe>
                <div class="fallback-content" id="iframe-fallback" style="display: none;">
                    <div class="text-center p-8">
                        <i class="fas fa-external-link-alt text-gray-400 text-4xl mb-4"></i>
                        <h4 class="text-xl font-bold text-gray-600 mb-2">Content Blocked</h4>
                        <p class="text-gray-500 mb-4">This content cannot be displayed in a frame</p>
                        <a href="${targetUrl}" target="_blank" class="btn-primary">
                            <i class="fas fa-external-link-alt mr-2"></i>Open in New Tab
                        </a>
                    </div>
                </div>
            `;
        }

        function handleIframeLoad(iframe) {
            // Iframe loaded successfully
        }

        function handleIframeError(iframe) {
            iframe.style.display = 'none';
            document.getElementById('iframe-fallback').style.display = 'flex';
        }

        function refreshPreview() {
            const url = document.getElementById('targetUrl').value.trim();
            if (url) {
                updateUserAgent();
                loadPreview(url);
            }
        }

        function stopPreview() {
            stopAutoReload();
            stopUptimeTimer();
            startTime = null;
            document.getElementById('statusText').textContent = 'Stopped | বন্ধ';
            document.getElementById('autoReload').checked = false;
            document.getElementById('reloadSettings').style.display = 'none';
        }

        function toggleAutoReload() {
            const autoReload = document.getElementById('autoReload');
            const reloadSettings = document.getElementById('reloadSettings');
            
            reloadSettings.style.display = autoReload.checked ? 'block' : 'none';
            
            if (!autoReload.checked) {
                stopAutoReload();
            } else {
                setupAutoReload();
            }
        }

        function setupAutoReload() {
            stopAutoReload();
            const interval = parseInt(document.getElementById('reloadInterval').value) * 1000;
            autoReloadTimer = setInterval(() => {
                const url = document.getElementById('targetUrl').value.trim();
                if (url && document.getElementById('autoReload').checked) {
                    updateUserAgent();
                    loadPreview(url);
                }
            }, interval);
        }

        function stopAutoReload() {
            if (autoReloadTimer) {
                clearInterval(autoReloadTimer);
                autoReloadTimer = null;
            }
        }

        function startUptimeTimer() {
            uptimeTimer = setInterval(() => {
                if (startTime) {
                    const uptime = Math.floor((Date.now() - startTime) / 1000);
                    document.getElementById('uptime').textContent = uptime + 's';
                }
            }, 1000);
        }

        function stopUptimeTimer() {
            if (uptimeTimer) {
                clearInterval(uptimeTimer);
                uptimeTimer = null;
            }
            document.getElementById('uptime').textContent = '0s';
        }

        function retryPreview() {
            const url = document.getElementById('targetUrl').value.trim();
            if (url) {
                loadPreview(url);
            }
        }

        function updateStats() {
            document.getElementById('loadCount').textContent = stats.loadCount;
            document.getElementById('successCount').textContent = stats.successCount;
            document.getElementById('errorCount').textContent = stats.errorCount;
        }

        function copyUserAgent() {
            navigator.clipboard.writeText(currentUserAgent).then(() => {
                showNotification('User-Agent copied | ইউজার-এজেন্ট কপি হয়েছে', 'success');
            }).catch(() => {
                // Fallback for older browsers
                const textArea = document.createElement('textarea');
                textArea.value = currentUserAgent;
                document.body.appendChild(textArea);
                textArea.select();
                document.execCommand('copy');
                document.body.removeChild(textArea);
                showNotification('User-Agent copied | ইউজার-এজেন্ট কপি হয়েছে', 'success');
            });
        }

        function openInNewTab() {
            const url = document.getElementById('targetUrl').value.trim();
            if (url && isValidUrl(url)) {
                window.open(url, '_blank');
            } else {
                showNotification('Please enter a valid URL | একটি বৈধ URL প্রবেশ করান', 'error');
            }
        }

        function isValidUrl(string) {
            try {
                new URL(string);
                return true;
            } catch (_) {
                return false;
            }
        }

        function showNotification(message, type) {
            const notification = document.createElement('div');
            notification.className = `notification ${type}`;
            notification.innerHTML = `
                <i class="fas ${type === 'success' ? 'fa-check-circle' : 'fa-exclamation-circle'} mr-2"></i>
                ${message}
            `;

            document.body.appendChild(notification);

            setTimeout(() => {
                notification.classList.add('show');
            }, 100);

            setTimeout(() => {
                notification.classList.remove('show');
                setTimeout(() => {
                    if (notification.parentNode) {
                        notification.parentNode.removeChild(notification);
                    }
                }, 300);
            }, 3000);
        }

        function saveSettings() {
            const settings = {
                targetUrl: document.getElementById('targetUrl').value,
                uaType: document.getElementById('uaType').value,
                customUserAgent: document.getElementById('customUserAgent').value,
                autoReload: document.getElementById('autoReload').checked,
                reloadInterval: document.getElementById('reloadInterval').value,
                currentMethod: currentMethod
            };
            localStorage.setItem('bloggerSmartToolSettings', JSON.stringify(settings));
        }

        function loadSettings() {
            const saved = localStorage.getItem('bloggerSmartToolSettings');
            if (saved) {
                try {
                    const settings = JSON.parse(saved);
                    if (settings.targetUrl) document.getElementById('targetUrl').value = settings.targetUrl;
                    if (settings.uaType) document.getElementById('uaType').value = settings.uaType;
                    if (settings.customUserAgent) document.getElementById('customUserAgent').value = settings.customUserAgent;
                    if (settings.autoReload) document.getElementById('autoReload').checked = settings.autoReload;
                    if (settings.reloadInterval) document.getElementById('reloadInterval').value = settings.reloadInterval;
                    if (settings.currentMethod) {
                        currentMethod = settings.currentMethod;
                        selectMethod(currentMethod);
                    }
                    
                    updateUserAgent();
                    toggleAutoReload();
                } catch (e) {
                    console.warn('Failed to load settings');
                }
            }
        }

        // Auto-save settings
        ['targetUrl', 'uaType', 'customUserAgent', 'autoReload', 'reloadInterval'].forEach(id => {
            const element = document.getElementById(id);
            if (element) {
                element.addEventListener('change', saveSettings);
                if (element.type === 'text' || element.type === 'url') {
                    element.addEventListener('input', saveSettings);
                }
            }
        });

        // Custom user agent input handler
        document.getElementById('customUserAgent').addEventListener('input', function() {
            if (document.getElementById('uaType').value === 'custom') {
                updateUserAgent();
            }
        });
    </script>
