        // Enhanced Platform Detection
        function detectCampaignType(url) {
            const urlLower = url.toLowerCase();
            
            // Social Media Platforms
            if (urlLower.includes('facebook.com') || urlLower.includes('fb.com') || urlLower.includes('fb.me')) return 'facebook';
            if (urlLower.includes('instagram.com') || urlLower.includes('instagr.am')) return 'instagram';
            if (urlLower.includes('youtube.com') || urlLower.includes('youtu.be')) return 'youtube';
            if (urlLower.includes('twitter.com') || urlLower.includes('t.co')) return 'twitter';
            if (urlLower.includes('tiktok.com') || urlLower.includes('vm.tiktok.com')) return 'tiktok';
            if (urlLower.includes('linkedin.com') || urlLower.includes('lnkd.in')) return 'linkedin';
            
            // Ad Networks
            if (urlLower.includes('adsterra')) return 'adsterra';
            if (urlLower.includes('monetag')) return 'monetag';
            
            // Campaign Types
            if (urlLower.includes('cpm') || urlLower.includes('rate')) return 'cpm';
            if (urlLower.includes('cpa') || urlLower.includes('action')) return 'cpa';
            if (urlLower.includes('affiliate') || urlLower.includes('ref=') || urlLower.includes('aff=')) return 'affiliate';
            if (urlLower.includes('shop') || urlLower.includes('buy') || urlLower.includes('product')) return 'ecommerce';
            
            // Default
            return 'website';
        }

        // Load Real Link in Device - THIS IS THE KEY FUNCTION FOR REAL LINK LOADING
        function loadRealLinkInDevice(device) {
            if (!currentActiveCampaign) return;
            const content = document.getElementById(device + 'Content');
            const overlay = document.getElementById(device + 'Overlay');

            // Remove existing content
            const existingIframe = content.querySelector('.link-preview-iframe');
            const existingCustom = content.querySelector('.custom-link-preview');
            if (existingIframe) existingIframe.remove();
            if (existingCustom) existingCustom.remove();

            // Try to load real link via iframe first
            const iframe = document.createElement('iframe');
            iframe.className = 'link-preview-iframe';
            iframe.src = currentActiveCampaign.url;
            iframe.style.transform = getDeviceScale(device);
            
            // Loading indicator
            const loadingDiv = document.createElement('div');
            loadingDiv.className = 'iframe-loading';
            loadingDiv.innerHTML = `
                <div class="loading-spinner" style="margin: 0 auto 4px;"></div>
                <div style="font-size: 10px;">Loading Real Link...</div>
            `;
            content.appendChild(loadingDiv);

            // Handle iframe load success
            iframe.onload = function() {
                loadingDiv.remove();
                overlay.classList.add('hidden');
                updateDeviceStatus(device, 'ACTIVE');
                currentActiveCampaign.realLinkLoaded = true;
                antiDetectionSystem.analytics.realLinksLoaded++;
                updateCampaignDisplay();
                updateAnalyticsDisplay();
                logActivity(`${getDeviceName(device)} real link loaded successfully: ${currentActiveCampaign.url}`, 'success');
            };

            // Handle iframe load error - fallback to custom preview
            iframe.onerror = function() {
                loadingDiv.remove();
                iframe.remove();
                createCustomLinkPreview(device);
            };

            // Timeout fallback for slow loading
            setTimeout(() => {
                if (content.contains(loadingDiv)) {
                    loadingDiv.remove();
                    iframe.remove();
                    createCustomLinkPreview(device);
                }
            }, 10000);

            content.appendChild(iframe);
        }

            logActivity('Real link loading system deactivated', 'info');
            document.getElementById('footerStatus').textContent = 'System Ready';
        }

        // Auto Traffic Generation with Real Link Loading
        function startAutoTraffic() {
            if (!antiDetectionSystem.isRunning || !currentActiveCampaign) return;
            const interval = parseInt(document.getElementById('reloadInterval').value) * 1000 || 15000;
            const randomDelay = parseInt(document.getElementById('randomDelay').value) * 1000 || 5000;
            const devices = Object.keys(antiDetectionSystem.devices);

            setTimeout(() => {
                if (antiDetectionSystem.isRunning && currentActiveCampaign) {
                    const randomDevice = devices[Math.floor(Math.random() * devices.length)];
                    generateRealLinkView(randomDevice);
                    startAutoTraffic();
                }
            }, interval + Math.random() * randomDelay);
        }

        // Generate Real Link View
        function generateRealLinkView(device) {
            if (!currentActiveCampaign) return;
            const deviceData = antiDetectionSystem.devices[device];

            currentActiveCampaign.views++;
            deviceData.views++;

            // Platform-specific earning calculation
            const platformMultiplier = getPlatformMultiplier(currentActiveCampaign.campaignType);
            const earningPerView = (0.001 + Math.random() * 0.004) * platformMultiplier;
            const deviceEarning = earningPerView;
            
            currentActiveCampaign.earnings += deviceEarning;
            deviceData.earnings += deviceEarning;

            antiDetectionSystem.analytics.totalViews++;
            antiDetectionSystem.analytics.totalEarnings += deviceEarning;

            updateDetectionRisk();
            updateDeviceDisplay(device);
            updateAnalyticsDisplay();
            updateCharts();

            if (Math.random() < 0.12) {
                reloadDevice(device);
            }

            logActivity(`${getDeviceName(device)} real link view: $${deviceEarning.toFixed(4)} [${getTypeTitle(currentActiveCampaign.campaignType)}]`, 'earning');
        }

        // Platform Earning Multipliers
        function getPlatformMultiplier(platform) {
            const multipliers = {
                'facebook': 1.3,
                'instagram': 1.2,
                'youtube': 1.5,
                'twitter': 1.1,
                'tiktok': 1.4,
                'linkedin': 1.6,
                'adsterra': 1.8,
                'monetag': 1.7,
                'cpm': 1.2,
                'cpa': 2.0,
                'affiliate': 1.5,
                'ecommerce': 1.3,
                'website': 1.0
            };
            return multipliers[platform] || 1.0;
        }

        // IP Rotation System
        function startIPRotation() {
            if (!antiDetectionSystem.isRunning) return;
            const rotationInterval = parseInt(document.getElementById('ipRotation').value) * 60000 || 600000;

            setTimeout(() => {
                if (antiDetectionSystem.isRunning) {
                    rotateAllIPs();
                    startIPRotation();
                }
            }, rotationInterval);
        }

        function rotateAllIPs() {
            const devices = Object.keys(antiDetectionSystem.devices);
            devices.forEach(device => {
                rotateDeviceIP(device);
            });
            showNotification('🔄 All IPs and User-Agents Rotated for Enhanced Security', 'info');
            logActivity('IP and User-Agent rotation completed for all devices - Real link loading maintained', 'info');
        }

        function rotateDeviceIP(device) {
            const deviceData = antiDetectionSystem.devices[device];
            const ipPool = ipPools[device];
            const uaPool = userAgentPools[device];

            const currentIPIndex = ipPool.indexOf(deviceData.ip);
            const nextIPIndex = (currentIPIndex + 1) % ipPool.length;
            deviceData.ip = ipPool[nextIPIndex];

            if (uaPool) {
                const currentUAIndex = uaPool.indexOf(deviceData.userAgent);
                const nextUAIndex = (currentUAIndex + 1) % uaPool.length;
                deviceData.userAgent = uaPool[nextUAIndex];
            }

            deviceData.lastRotation = new Date().toLocaleTimeString();
            updateDeviceIPDisplay(device);
            updateDetectionRisk();
        }

        // Update Device IP Display
        function updateDeviceIPDisplay(device) {
            const deviceData = antiDetectionSystem.devices[device];

            const ipElement = document.getElementById(device + 'IPDisplay');
            if (ipElement) {
                ipElement.textContent = `IP: ${deviceData.ip}`;
            }

            const uaElement = document.getElementById(device + 'UADisplay');
            if (uaElement) {
                const shortUA = deviceData.userAgent.split(' ')[0] + '/' + deviceData.userAgent.split('/').pop().split(' ')[0];
                uaElement.textContent = shortUA;
            }

            const lastRotationElement = document.getElementById(device + 'LastRotation');
            if (lastRotationElement) {
                lastRotationElement.textContent = deviceData.lastRotation || '-';
            }
        }

        // Update Detection Risk
        function updateDetectionRisk() {
            const totalViews = antiDetectionSystem.analytics.totalViews;
            const linksLoaded = antiDetectionSystem.analytics.realLinksLoaded;
            let risk = 'Very Low';
            let riskClass = 'risk-very-low';

            if (totalViews > 1000 && linksLoaded < 5) {
                risk = 'Low';
                riskClass = 'risk-low';
            } else if (totalViews > 5000 && linksLoaded < 10) {
                risk = 'Medium';
                riskClass = 'risk-medium';
            } else if (totalViews > 10000 && linksLoaded < 20) {
                risk = 'High';
                riskClass = 'risk-high';
            }

            antiDetectionSystem.analytics.detectionRisk = risk;
            const riskElement = document.getElementById('detectionRisk');
            if (riskElement) {
                riskElement.textContent = risk;
                riskElement.className = 'risk-indicator ' + riskClass;
            }
        }

        // Reload Device with Real Link
        function reloadDevice(device) {
            if (!currentActiveCampaign) return;
            updateDeviceStatus(device, 'ROTATING');
            setTimeout(() => {
                rotateDeviceIP(device);
                startRealLinkLoadingOnDevice(device);
            }, 1500 + Math.random() * 2500);
        }

        // Update Device Display
        function updateDeviceDisplay(device) {
            const deviceData = antiDetectionSystem.devices[device];

            const viewElement = document.getElementById(device + 'Views');
            if (viewElement) {
                viewElement.textContent = deviceData.views.toLocaleString();
            }

            const earningElement = document.getElementById(device + 'Earning');
            if (earningElement) {
                earningElement.textContent = '$' + deviceData.earnings.toFixed(2);
            }
        }

        function updateDeviceStatus(device, status) {
            const statusElement = document.getElementById(device + 'Status');
            if (statusElement) {
                statusElement.textContent = status;
                statusElement.className = 'device-status ' + (status === 'READY' ? 'status-ready' : status === 'LOADING' ? 'status-loading' : status === 'ACTIVE' ? 'status-active' : status === 'ROTATING' ? 'status-rotating' : 'status-ready');
            }
        }

        // Update Analytics Display
        function updateAnalyticsDisplay() {
            document.getElementById('totalCampaigns').textContent = campaigns.length;
            document.getElementById('totalTrafficViews').textContent = antiDetectionSystem.analytics.totalViews.toLocaleString();
            document.getElementById('totalEarnings').textContent = '$' + antiDetectionSystem.analytics.totalEarnings.toFixed(2);
            document.getElementById('activeDevicesCount').textContent = antiDetectionSystem.analytics.activeDevices;
            document.getElementById('totalUniqueViews').textContent = antiDetectionSystem.analytics.totalViews.toLocaleString();
            document.getElementById('uniqueIPs').textContent = antiDetectionSystem.analytics.uniqueIPs;
            document.getElementById('realLinksLoaded').textContent = antiDetectionSystem.analytics.realLinksLoaded;
            document.getElementById('naturalPattern').textContent = antiDetectionSystem.analytics.naturalPattern + '%';
        }

        // Update Charts with Enhanced Containment
        function updateCharts() {
            if (!deviceDistributionChart || !trafficChart || !deviceChart) return;

            try {
                const newTrafficData = Array(12).fill(0).map(() => Math.floor(Math.random() * 100) + 20);
                trafficChart.data.datasets[0].data = newTrafficData;
                trafficChart.update('none');

                const deviceData = Object.values(antiDetectionSystem.devices).map(device => device.views);
                const total = deviceData.reduce((sum, val) => sum + val, 0);
                if (total > 0) {
                    deviceChart.data.datasets[0].data = deviceData;
                    deviceChart.update('none');
                }
            } catch (error) {
                console.warn('Chart update failed:', error);
            }
        }

        // Additional Control Functions
        function previewAllDevices() {
            if (!currentActiveCampaign) {
                showNotification('Please select a campaign first', 'error');
                return;
            }
            startRealLinkLoadingOnAllDevices();
            showNotification('👁️ Loading real ' + getTypeTitle(currentActiveCampaign.campaignType) + ' links on all devices...', 'info');
        }

        function generateMassTraffic() {
            if (!currentActiveCampaign) {
                showNotification('Please select a campaign first', 'error');
                return;
            }
            showNotification('📈 Mass traffic generation started for ' + getTypeTitle(currentActiveCampaign.campaignType) + ' with real link loading...', 'info');
            const devices = Object.keys(antiDetectionSystem.devices);
            const massTrafficInterval = setInterval(() => {
                const randomDevice = devices[Math.floor(Math.random() * devices.length)];
                generateRealLinkView(randomDevice);
            }, 50 + Math.random() * 200);

            setTimeout(() => {
                clearInterval(massTrafficInterval);
                showNotification('✅ Mass traffic generation completed successfully with real link loading', 'success');
            }, 25000);
        }

        function resetAntiDetection() {
            if (confirm('Reset all real link loading data and statistics?')) {
                stopAntiDetectionSystem();
                Object.keys(antiDetectionSystem.devices).forEach(device => {
                    antiDetectionSystem.devices[device].views = 0;
                    antiDetectionSystem.devices[device].earnings = 0;
                    antiDetectionSystem.devices[device].lastRotation = null;
                    antiDetectionSystem.devices[device].status = 'ready';
                    const viewElement = document.getElementById(device + 'Views');
                    if (viewElement) viewElement.textContent = '0';
                    const earningElement = document.getElementById(device + 'Earning');
                    if (earningElement) earningElement.textContent = '$0';
                    updateDeviceStatus(device, 'READY');
                    
                    // Remove any existing real link content
                    const content = document.getElementById(device + 'Content');
                    const existingIframe = content.querySelector('.link-preview-iframe');
                    const existingCustom = content.querySelector('.custom-link-preview');
                    if (existingIframe) existingIframe.remove();
                    if (existingCustom) existingCustom.remove();
                    
                    // Show overlay again
                    const overlay = document.getElementById(device + 'Overlay');
                    if (overlay) {
                        overlay.classList.remove('hidden');
                    }
                });

                antiDetectionSystem.analytics.totalViews = 0;
                antiDetectionSystem.analytics.totalEarnings = 0;
                antiDetectionSystem.analytics.realLinksLoaded = 0;
                antiDetectionSystem.analytics.activeDevices = 0;

                campaigns.forEach(campaign => {
                    campaign.views = 0;
                    campaign.earnings = 0;
                    campaign.realLinkLoaded = false;
                });

                updateCampaignDisplay();
                updateAnalyticsDisplay();
                updateCharts();
                showNotification('🔄 Real link loading system and all data reset successfully', 'success');
                logActivity('Complete real link loading system and statistical data reset performed', 'info');
            }
        }

        // Utility Functions
        function isValidUrl(string) {
            try {
                new URL(string);
                return true;
            } catch (_) {
                return false;
            }
        }

        function deleteCampaign(campaignId) {
            if (confirm('Delete this campaign permanently?')) {
                const campaign = campaigns.find(c => c.id == campaignId);
                campaigns = campaigns.filter(c => c.id != campaignId);
                if (currentActiveCampaign && currentActiveCampaign.id == campaignId) {
                    currentActiveCampaign = null;
                }
                updateCampaignDisplay();
                updateAnalyticsDisplay();
                showNotification('🗑️ Campaign deleted successfully', 'success');
                logActivity('Campaign deleted: ' + (campaign ? campaign.title : 'Unknown'), 'info');
            }
        }

        function getDeviceName(device) {
            const names = {
                'android': 'Android',
                'iphone': 'iPhone',
                'samsung': 'Samsung',
                'ipad': 'iPad',
                'windows': 'Windows',
                'mac': 'Mac'
            };
            return names[device] || device;
        }

        // Enhanced Notification System
        function showNotification(message, type) {
            const notification = document.createElement('div');
            notification.className = 'notification ' + type;
            notification.innerHTML = `
                <i class="fas ${type === 'success' ? 'fa-check-circle' : type === 'error' ? 'fa-exclamation-triangle' : type === 'warning' ? 'fa-exclamation-circle' : 'fa-info-circle'} mr-2"></i>
                ${message}
            `;
            document.body.appendChild(notification);
            setTimeout(() => notification.classList.add('show'), 100);
            setTimeout(() => {
                notification.classList.remove('show');
                setTimeout(() => {
                    if (notification.parentNode) {
                        notification.parentNode.removeChild(notification);
                    }
                }, 300);
            }, 6000);
        }

        // Enhanced Activity Logging
        function logActivity(message, type) {
            const logContainer = document.getElementById('activityLog');
            const activityElement = document.createElement('div');
            activityElement.className = `activity-item`;
            activityElement.innerHTML = `
                <div class="flex items-center justify-between">
                    <div class="flex items-center gap-2">
                        <div class="w-2 h-2 ${type === 'success' ? 'bg-green-500' : type === 'info' ? 'bg-blue-500' : type === 'warning' ? 'bg-yellow-500' : type === 'earning' ? 'bg-purple-500' : 'bg-red-500'} rounded-full"></div>
                        <span class="text-sm">[${new Date().toLocaleTimeString()}] ${message}</span>
                    </div>
                </div>
            `;
            logContainer.insertBefore(activityElement, logContainer.firstChild);

            const activities = logContainer.children;
            if (activities.length > 15) {
                logContainer.removeChild(activities[activities.length - 1]);
            }
        }

        // Data Management
        function saveData() {
            const data = {
                campaigns: campaigns,
                antiDetectionSystem: antiDetectionSystem,
                version: 'v16.0',
                lastSaved: new Date().toISOString()
            };
            localStorage.setItem('realLinkLoadingCampaignSystemV16', JSON.stringify(data));
        }

        function loadSavedData() {
            const saved = localStorage.getItem('realLinkLoadingCampaignSystemV16');
            if (saved) {
                try {
                    const data = JSON.parse(saved);
                    if (data.campaigns) {
                        campaigns = data.campaigns;
                        updateCampaignDisplay();
                    }
                    if (data.antiDetectionSystem) {
                        antiDetectionSystem = Object.assign(antiDetectionSystem, data.antiDetectionSystem);
                        Object.keys(antiDetectionSystem.devices).forEach(device => {
                            updateDeviceDisplay(device);
                            updateDeviceIPDisplay(device);
                        });
                    }
                    updateAnalyticsDisplay();
                    logActivity('Previous real link loading session data loaded successfully', 'info');
                } catch (e) {
                    console.warn('Failed to load saved data:', e);
                }
            }
        }
