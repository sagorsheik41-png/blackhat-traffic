  <script>
    (function(){
      // Mock server response for demonstration purposes
      const mockResponse = {
        ok: true,
        images: [
          "https://placehold.co/600x400/FF0000/FFFFFF?text=FB+Image+1",
          "https://placehold.co/600x400/00FF00/FFFFFF?text=FB+Image+2",
          "https://placehold.co/600x400/0000FF/FFFFFF?text=FB+Image+3"
        ],
        videos: [
          "https://sample-videos.com/zip/10/mp4/SampleVideo_1280x720_1mb.mp4"
        ]
      };

      const btn = document.getElementById('btn-extract');
      const urlInput = document.getElementById('fb-url');
      const resBox = document.getElementById('fb-result');
      const imagesDiv = document.getElementById('images');
      const videosDiv = document.getElementById('videos');
      const msg = document.getElementById('fb-msg');
      const clearBtn = document.getElementById('btn-clear');

      clearBtn.onclick = () => {
        urlInput.value = '';
        imagesDiv.innerHTML = '';
        videosDiv.innerHTML = '';
        resBox.style.display = 'none';
        msg.textContent = '';
      };

      btn.onclick = async () => {
        msg.textContent = '';
        imagesDiv.innerHTML = '';
        videosDiv.innerHTML = '';
        resBox.style.display = 'none';

        let url = urlInput.value.trim();
        if (!url) { 
          msg.textContent = 'Please paste a Facebook profile or post URL.'; 
          return; 
        }

        // Show loading state
        btn.innerHTML = '<span class="loading"></span>Processing...';
        btn.disabled = true;

        try {
          // Simulate API call delay
          await new Promise(resolve => setTimeout(resolve, 1500));
          
          // In a real implementation, you would make an API call to your server
          // const resp = await fetch((server.replace(/\/$/, '')) + '/api/extract', {
          //   method: 'POST',
          //   headers: { 'Content-Type': 'application/json' },
          //   body: JSON.stringify({ url })
          // });
          // const data = await resp.json();
          
          // Using mock response for demo
          const data = mockResponse;
          
          if (!data.ok) {
            msg.textContent = data.error || 'No media found or the post is private.';
            return;
          }
          
          const { images = [], videos = [] } = data;
          if (images.length === 0 && videos.length === 0) {
            msg.textContent = 'No downloadable media found. Try a different post or public profile.';
            return;
          }
          
          // Show images
          images.forEach(src => {
            const card = document.createElement('div');
            card.className = 'media-card';
            card.innerHTML = `
              <a target="_blank" href="${src}">
                <img src="${src}" alt="image">
              </a>
              <div class="download-link">
                <a download href="${src}">Download</a>
              </div>
            `;
            imagesDiv.appendChild(card);
          });

          // Show videos
          videos.forEach(src => {
            const wrap = document.createElement('div');
            wrap.className = 'video-wrapper';
            wrap.innerHTML = `
              <video controls>
                <source src="${src}" type="video/mp4">
                Your browser does not support the video tag.
              </video>
              <div class="video-links">
                <a target="_blank" href="${src}">Open video</a> • 
                <a download href="${src}">Download</a>
              </div>
            `;
            videosDiv.appendChild(wrap);
          });

          resBox.style.display = 'block';
        } catch (err) {
          console.error(err);
          msg.textContent = 'An error occurred while processing your request.';
        } finally {
          // Reset button
          btn.innerHTML = 'Extract';
          btn.disabled = false;
        }
      };
    })();
  </script>
