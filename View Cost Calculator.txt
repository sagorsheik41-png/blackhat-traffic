  <script>
    function calculate() {
      const costPerView = parseFloat(document.getElementById('costPerView').value);
      const views = parseInt(document.getElementById('views').value);
      const timePerView = parseInt(document.getElementById('timePerView').value);
      const rateBDT = parseFloat(document.getElementById('rateBDT').value);

      if (isNaN(costPerView) || isNaN(views) || isNaN(timePerView) || isNaN(rateBDT) || views < 1 || costPerView < 0 || rateBDT < 0) {
        document.getElementById('rublesResult').textContent = `Invalid input. Please check the numbers.`;
        document.getElementById('bdtResult').textContent = ``;
        document.getElementById('totalTime').textContent = ``;
        document.getElementById('costPerMinute').textContent = ``;
        document.getElementById('result').classList.remove('hidden');
        return;
      }

      const totalRubles = costPerView * views;
      const totalBDT = totalRubles * rateBDT;
      const totalSeconds = timePerView * views;
      const hours = Math.floor(totalSeconds / 3600);
      const minutes = Math.floor((totalSeconds % 3600) / 60);
      const seconds = totalSeconds % 60;
      const totalMinutes = totalSeconds / 60;
      const costPerMinute = totalMinutes > 0 ? totalRubles / totalMinutes : 0; 

      // Display Results with comma formatting
      document.getElementById('rublesResult').innerHTML = `<span class="text-purple-700">Total Cost:</span> ${totalRubles.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })} rubles.`;
      document.getElementById('bdtResult').innerHTML = `<span class="text-purple-700">Equivalent:</span> ${totalBDT.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })} ৳ (BDT).`;
      document.getElementById('totalTime').innerHTML = `<span class="text-purple-700">Total Viewing Time:</span> ${hours} hours ${minutes} minutes ${seconds} seconds.`;
      document.getElementById('costPerMinute').innerHTML = `<span class="text-purple-700">Avg. Cost per Minute:</span> ${costPerMinute.toLocaleString('en-US', { minimumFractionDigits: 4, maximumFractionDigits: 4 })} rubles.`;

      document.getElementById('result').classList.remove('hidden');
      
      // Scroll to the result section on mobile after calculation
      if (window.innerWidth < 640) {
        document.getElementById('result').scrollIntoView({ behavior: 'smooth', block: 'start' });
      }
    }

    // --- UX Enhancements ---
    // 1. Auto-calculate on input change
    document.querySelectorAll('input, select').forEach(input => {
      input.addEventListener('input', calculate);
    });

    // 2. Run calculation once on page load to show default results instantly
    window.addEventListener('load', calculate);
  </script>
