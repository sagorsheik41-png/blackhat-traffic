        function logDebug(message) {
            const debugLog = document.getElementById('debugLog');
            const messages = debugLog.getElementsByTagName('p');
            if (messages.length >= maxDebugMessages) {
                messages[0].remove(); // Remove oldest message

        function handleGameMessage(data) {
            if (isSimulating && isConnected) {
                isSimulating = false;
                logDebug('Real WebSocket connected, stopping simulation');
            }
            logDebug(`Received: ${JSON.stringify(data)}`);
            const now = Date.now();
            gameState.lastUpdate = now;

            switch (data.type) {
                case 'crazytime.newGame':
                    gameState.gameId = data.args.gameId || '';
                    gameState.gameNumber = data.args.gameNumber || '';
                    gameState.status = 'New Game';
                    gameState.betsOpenTime = 0;
                    gameState.resultTime = 0;
                    clearSignalBoxes();
                    setTimeout(() => {
                        if (gameState.spinHistory.length > 0) {
                            showEarlyPrediction();
                        }
                    }, 1000);
                    updateGameDisplay();
                    break;
                case 'crazytime.betsOpen':
                    gameState.status = 'Bets Open';
                    gameState.betsOpenTime = now;
                    startPredictionLoop();
                    updateGameDisplay();
                    break;
                case 'crazytime.betsClosed':
                    gameState.status = 'Bets Closed';
                    updateGameDisplay();
                    break;
                case 'crazytime.result':
                    gameState.result = data.args.result || '';
                    gameState.multiplier = data.args.totalMultiplier || 0;
                    gameState.status = 'Result';
                    gameState.resultTime = now;
                    if (gameState.betsOpenTime > 0) {
                        roundDuration = Math.max(20000, Math.min(60000, now - gameState.betsOpenTime));
                        logDebug(`Updated round duration to ${roundDuration}ms`);
                    }
                    updateGameDisplay();
                    break;
                case 'crazytime.gameWinners':
                    gameState.winners = data.args.winners || [];
                    gameState.totalWinners = data.args.totalWinners || 0;
                    gameState.totalAmount = data.args.totalAmount || 0;
                    updateWinnersDisplay();
                    break;
                case 'crazytime.spinHistory':
                    gameState.spinHistory = data.args.results || [];
                    updateSpinHistory();
                    break;
                case 'crazytime.slot.result':
                    gameState.result = data.args.result || '';
                    gameState.multiplier = data.args.multiplier || 0;
                    gameState.status = 'Slot Win';
                    updateGameDisplay();
                    break;

                    gameState.bonusResult = data.args.flappers ?

                    break;

            }
        }

        function predictNextSignal() {
            if (gameState.spinHistory.length > 0 && gameState.betsOpenTime > 0) {
                logDebug(`Predicting with ${gameState.spinHistory.length} history entries`);
                const resultCount = {};
                const recentHistory = gameState.spinHistory.slice(-10);
                const probabilities = { '1': 0.28, '2': 0.26, '5': 0.13, '10': 0.07, 'b1': 0.13, 'b2': 0.07, 'b3': 0.04, 'b4': 0.02 }; // Based on Crazy Time wheel segments

                // Weight recent results with base probabilities
                recentHistory.forEach((result, index) => {
                    const weight = Math.max(1, 10 - index);
                    resultCount[result.result] = (resultCount[result.result] || 0) + weight * (probabilities[result.result] || 0.1);
                });

                const patterns = analyzePatterns(recentHistory);
                if (patterns.mostLikely) {
                    resultCount[patterns.mostLikely] = (resultCount[patterns.mostLikely] || 0) + 15;
                }

                let mostFrequent = '';
                let maxCount = 0;
                for (const result in resultCount) {
                    if (resultCount[result] > maxCount) {
                        maxCount = resultCount[result];
                        mostFrequent = result;
                    }
                }

                const confidence = Math.min(95, (maxCount / recentHistory.length * 100)).toFixed(1);
                const timeIntoRound = Date.now() - gameState.betsOpenTime;
                const remainingTime = roundDuration - timeIntoRound;
                const signalDisplayLeadTime = 15000;

                if (remainingTime > signalDisplayLeadTime && mostFrequent) {
                    predictedSignal = mostFrequent;
                    activateSignalBox(predictedSignal);
                    const timeToResult = Math.floor(remainingTime / 1000);
                    document.getElementById('nextRoundPrediction').textContent = `Betting Signal: ${predictedSignal} likely (Confidence: ${confidence}%)`;
                    document.getElementById('earlyBettingSignal').textContent = `🎯 NEXT ROUND: ${predictedSignal} - Place bet now! (${timeToResult}s remaining)`;
                    logDebug(`Predicted ${predictedSignal} with confidence ${confidence}% - ${timeToResult}s remaining`);
                } else if (predictedSignal) {
                    activateSignalBox(predictedSignal);
                    document.getElementById('nextRoundPrediction').textContent = `Betting Signal: ${predictedSignal} likely (Confidence: ${confidence}%)`;
                    document.getElementById('earlyBettingSignal').textContent = `🎯 NEXT ROUND: ${predictedSignal} - Get ready!`;
                } else {
                    document.getElementById('nextRoundPrediction').textContent = 'Betting Signal: Analyzing patterns...';
                    document.getElementById('earlyBettingSignal').textContent = '';
                    clearSignalBoxes();
                    logDebug('No prediction: Insufficient data or signal window passed');
                }
            } else {
                document.getElementById('nextRoundPrediction').textContent = 'Betting Signal: Waiting for game data...';
                document.getElementById('earlyBettingSignal').textContent = '';
                clearSignalBoxes();

            }
        }

        function analyzePatterns(history) {
            if (history.length < 3) return { mostLikely: null };
            const last3 = history.slice(-3).map(r => r.result);
            const numbers = ['1', '2', '5', '10'];
            const bonuses = ['b1', 'b2', 'b3', 'b4'];
            const recentNumbers = last3.filter(r => numbers.includes(r)).length;
            const recentBonuses = last3.filter(r => bonuses.includes(r)).length;

            if (recentNumbers >= 2 && recentBonuses === 0) {
                return { mostLikely: 'b1' };
            }
            if (recentBonuses >= 2 && recentNumbers === 0) {
                return { mostLikely: '2' };
            }
            const last2 = history.slice(-2).map(r => r.result);
            if (last2[0] === last2[1]) {
                const different = numbers.concat(bonuses).filter(r => r !== last2[0]);
                return { mostLikely: different[Math.floor(Math.random() * different.length)] };
            }
            return { mostLikely: null };
        }

        function startPredictionLoop() {
            if (predictionInterval) clearInterval(predictionInterval);
            predictionInterval = setInterval(() => {

            logDebug('Prediction loop started');
        }

        function validateWebSocketUrl(url) {
            try {
                new URL(url);
                return url.startsWith('wss://');
            } catch {
                return false;
            }
        }

        function connectWebSocket() {
            const wsUrlInput = document.getElementById('wsUrl').value;
            if (!validateWebSocketUrl(wsUrlInput)) {
                document.getElementById('connectionStatus').textContent = 'Invalid WebSocket URL';
                document.getElementById('connectionStatus').classList.add('text-red-500');
                logDebug('Invalid WebSocket URL provided');
                return;
            }

            if (ws && isConnected) {
                ws.close();
            }
            ws = new WebSocket(wsUrlInput);
            isConnected = false;
            reconnectAttempts = 0;
            document.getElementById('connectionStatus').textContent = 'Connecting...';
            document.getElementById('connectionStatus').classList.remove('text-green-500', 'text-red-500');
            logDebug(`Attempting connection to ${wsUrlInput}`);

            ws.onopen = () => {
                logDebug('Connected to WebSocket server');
                isConnected = true;
                isSimulating = false;
                reconnectAttempts = 0;
                document.getElementById('connectionStatus').textContent = 'Connected (Live)';
                document.getElementById('connectionStatus').classList.add('text-green-500');
                ws.send(JSON.stringify({ subscribe: { channel: 'CrazyTime0000001' } }));
            };

            ws.onmessage = (event) => {
                try {
                    const data = JSON.parse(event.data);
                    handleGameMessage(data);
                } catch (error) {
                    logDebug(`Error parsing message: ${error.message}`);
                }
            };

            ws.onerror = (error) => {
                logDebug(`WebSocket error: ${JSON.stringify(error)}`);
                document.getElementById('connectionStatus').textContent = 'Connection error';
                document.getElementById('connectionStatus').classList.add('text-red-500');
                reconnectWebSocket();
            };

            ws.onclose = () => {
                logDebug('Disconnected. Attempting to reconnect...');
                isConnected = false;
                document.getElementById('connectionStatus').textContent = 'Disconnected';
                document.getElementById('connectionStatus').classList.add('text-red-500');
                reconnectWebSocket();
            };
        }

        function reconnectWebSocket() {
            if (reconnectAttempts < maxReconnectAttempts) {
                const delay = Math.min(1000 * Math.pow(2, reconnectAttempts), 30000); // Exponential backoff
                reconnectAttempts++;
                setTimeout(connectWebSocket, delay);
                logDebug(`Reconnect attempt ${reconnectAttempts}/${maxReconnectAttempts} in ${delay}ms`);
            } else {
                logDebug('Max reconnect attempts reached. Starting simulation...');
                startSimulation();
            }
        }

        function startSimulation() {
            if (!isConnected && !isSimulating) {
                isSimulating = true;
                document.getElementById('connectionStatus').textContent = 'Disconnected (Simulation Mode)';
                document.getElementById('connectionStatus').classList.add('text-yellow-500');
                logDebug('Starting simulation with sample data');
                simulatedData.forEach((data, index) => {
                    setTimeout(() => {
                        if (!isConnected) handleGameMessage(data);
                    }, index * 7000);
                });
            }
        }
