        // Signal timer
        function startAdvancedSignalTimer() {
            const timerContainer = document.getElementById('signalTimerContainer');
            timerContainer.classList.add('active');

            signalTimer = setInterval(() => {
                const timeLeft = Math.max(0, activeSignal.endTime - Date.now());
                const totalSeconds = Math.floor(timeLeft / 1000);
                const minutes = Math.floor(totalSeconds / 60);
                const seconds = totalSeconds % 60;
                document.getElementById('signalTimer').textContent = `${minutes}:${seconds.toString().padStart(2, '0')}`;
                if (timeLeft <= 0) {
                    clearInterval(signalTimer);
                    timerContainer.classList.remove('active');
                    processAdvancedSignalResult();
                }
            }, 1000);
        }

        // Signal result processing
        function processAdvancedSignalResult() {
            if (!activeSignal) return;
            const platformConfig = platformConfigs[activeSignal.platform];
            let winRate = platformConfig.winRate;

            if (activeSignal.isPremium) {
                const planConfig = premiumPlans[currentUser.premiumSubscription.plan];
                winRate = Math.max(winRate, 0.87);
                if (currentUser.premiumSubscription.plan === 'pro') winRate = Math.max(winRate, 0.92);
                if (currentUser.premiumSubscription.plan === 'vip') winRate = Math.max(winRate, 0.96);
            }
            const adjustedWinRate = winRate + (activeSignal.confidence - 90) * 0.01;
            const isWin = Math.random() < adjustedWinRate;

            let baseProfit = 50 + Math.random() * 200;
            if (activeSignal.duration >= 300) baseProfit *= 1.5;
            if (activeSignal.isPremium) baseProfit *= 1.3;

            const profit = isWin ? baseProfit : -baseProfit * 0.8;
            const trade = {
                time: new Date().toISOString(),
                pair: activeSignal.pair,
                platform: activeSignal.platform,
                direction: activeSignal.direction,
                duration: activeSignal.duration,
                result: isWin ? 'WIN' : 'LOSS',
                profit: profit,
                isPremium: activeSignal.isPremium
            };

            currentUser.totalTrades++;
            if (isWin) currentUser.winningTrades++;
            currentUser.totalProfit += profit;
            if (!currentUser.trades) currentUser.trades = [];
            currentUser.trades.unshift(trade);
            currentUser.trades = currentUser.trades.slice(0, 50);
            users[currentUser.email] = currentUser;
            saveData();
            showProfitAnimation(isWin, Math.abs(profit));
            
            const wasPremiumSignal = activeSignal.isPremium;
            activeSignal = null;

            document.getElementById('signalStatus').textContent = 'Ready to Generate';
            document.getElementById('generateBtn').disabled = false;
            document.getElementById('generateBtn').innerHTML = '<i class="fas fa-brain mr-2"></i>Generate Advanced AI Signal';
            updateSignalDisplay();

            showNotification(
                isWin ? `✅ ${wasPremiumSignal ? '🔒 ' : ''}Trade Won! +$${profit.toFixed(2)}` : `❌ ${wasPremiumSignal ? '🔒 ' : ''}Trade Lost! -$${Math.abs(profit).toFixed(2)}`,
                isWin ? 'success' : 'error'
            );
            renderUserDashboard();
        }

        // Chart Initialization
        function initPerformanceChart() {
            const ctx = document.getElementById('performanceChart'); if (!ctx || !currentUser) return;
            if (performanceChart) performanceChart.destroy();
            const profits = (currentUser.trades || []).slice(0, 15).reverse().map(t => t.profit);
            const cumulativeProfit = profits.reduce((acc, p) => [...acc, (acc.length > 0 ? acc[acc.length - 1] : 0) + p], []);
            performanceChart = new Chart(ctx, {
                type: 'line', data: { labels: Array(cumulativeProfit.length).fill(''), datasets: [{ label: 'Profit Growth ($)', data: cumulativeProfit, borderColor: '#10b981', backgroundColor: 'rgba(16, 185, 129, 0.1)', tension: 0.4, fill: true }] },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { x: { ticks: { display: false }, grid: { display: false } }, y: { ticks: { color: '#9ca3af', font: { size: 10 } }, grid: { color: 'rgba(156, 163, 175, 0.2)' } } } }
            });
        }
        function initRevenueChart() {
            const ctx = document.getElementById('revenueChart'); if (!ctx) return;
            if (revenueChart) revenueChart.destroy();
            const planCounts = { 'Premium Basic': 0, 'Premium Pro': 0, 'Premium VIP': 0 };
            Object.values(users).forEach(u => { if (u.premiumSubscription && u.premiumSubscription.status === 'active') { const p = `Premium ${u.premiumSubscription.plan.charAt(0).toUpperCase() + u.premiumSubscription.plan.slice(1)}`; if(planCounts.hasOwnProperty(p)) planCounts[p]++; } });
            revenueChart = new Chart(ctx, {
                type: 'doughnut', data: { labels: Object.keys(planCounts), datasets: [{ data: Object.values(planCounts), backgroundColor: ['#3b82f6', '#8b5cf6', '#f59e0b'], borderWidth: 0 }] },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { position: 'bottom', labels: { color: '#fff' } } } }
            });
        }

            const notification = document.getElementById('notification');
            notification.innerHTML = message;
            notification.className = `notification show ${type === 'success' ? 'bg-green-600' : type === 'error' ? 'bg-red-600' : type === 'warning' ? 'bg-yellow-600' : 'bg-blue-600'}`;
            setTimeout(() => { notification.classList.remove('show'); }, 5000);
        }
