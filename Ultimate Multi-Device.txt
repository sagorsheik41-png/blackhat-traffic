        function startDeviceTimer(device) {
            const state = appState.deviceStates[device];
            if (state.timer) clearInterval(state.timer);
            
            state.active = true;
            state.counter = appState.deviceTimers[device];
            
            updateDeviceCounter(device, state.counter);
            
            state.timer = setInterval(() => {
                state.counter--;
                updateDeviceCounter(device, state.counter);
                
                if (state.counter <= 0) {
                    if (appState.urls.length > 0) {
                        loadUrlInDevice(device);
                    }
                    state.counter = appState.deviceTimers[device];
                }
            }, 1000);

        function simulateVisit(device, url = null) {
            if (!url && appState.urls.length === 0) {
                showNotification('প্রথমে URL যোগ করুন', 'error');
                return;
            }
            
            const targetUrl = url || appState.urls[Math.floor(Math.random() * appState.urls.length)];
            const userAgent = appState.userAgents[device];
            const geoLocation = getRandomGeoLocation();
            
            // Simulate realistic visitor behavior
            targetUrl.visitors++;
            targetUrl.impressions += Math.floor(Math.random() * 3) + 1;
            targetUrl.deviceStats[device]++;
            
            // Simulate click with realistic CTR
            if (Math.random() > 0.75) {
                targetUrl.clicks++;
            }
            
            // Store visitor data
            if (!targetUrl.userAgents.includes(userAgent)) {
                targetUrl.userAgents.push(userAgent);
            }
            
            if (!targetUrl.geoStats[geoLocation.country]) {
                targetUrl.geoStats[geoLocation.country] = 0;
            }
            targetUrl.geoStats[geoLocation.country]++;
            
            targetUrl.lastActivity = new Date();
            targetUrl.conversionRate = Math.min(targetUrl.conversionRate + 0.1, 25);
            
            updateAnalytics();
            updateUrlsDisplay();
            
            showNotification(`✅ ${device} ডিভাইস থেকে অর্গানিক ভিজিট সিমুলেট করা হয়েছে`, 'success');
            logActivity(`✅ Simulated organic visit from ${device} device for ${targetUrl.title}`, 'success');
        }

        function generateAutoTraffic() {
            if (!appState.trafficGeneration.active || appState.urls.length === 0) return;
            
            const randomUrl = appState.urls[Math.floor(Math.random() * appState.urls.length)];
            const devices = ['desktop', 'tablet', 'android', 'iphone'];
            const randomDevice = devices[Math.floor(Math.random() * devices.length)];
            const userAgent = appState.userAgents[randomDevice];
            const geoLocation = getRandomGeoLocation();
            
            // Simulate realistic traffic patterns
            const trafficMultiplier = getTrafficMultiplier();
            const visitorsToGenerate = Math.floor((appState.trafficGeneration.rate / 720) * trafficMultiplier);
            
            for (let i = 0; i < visitorsToGenerate; i++) {
                randomUrl.visitors++;
                randomUrl.impressions += Math.floor(Math.random() * 4) + 1;
                randomUrl.deviceStats[randomDevice]++;
                
                // Realistic click-through rate
                if (Math.random() > 0.85) {
                    randomUrl.clicks++;
                }
                
                // Store unique user agents and geo data
                if (!randomUrl.userAgents.includes(userAgent)) {
                    randomUrl.userAgents.push(userAgent);
                }
                
                if (!randomUrl.geoStats[geoLocation.country]) {
                    randomUrl.geoStats[geoLocation.country] = 0;
                }
                randomUrl.geoStats[geoLocation.country]++;
            }
            
            randomUrl.lastActivity = new Date();
            updateAnalytics();
            
            if (Math.random() > 0.8) {
                logActivity(`📈 Auto-generated ${visitorsToGenerate} organic visits for ${randomUrl.title} on ${randomDevice}`, 'info');
            }
        }

        function getTrafficMultiplier() {
            const pattern = document.getElementById('trafficPattern').value;
            const hour = new Date().getHours();
            
            switch (pattern) {
                case 'viral':
                    return Math.random() * 3 + 1;
                case 'organic':
                    if (hour >= 9 && hour <= 17) return 1.5;
                    if (hour >= 19 && hour <= 22) return 1.2;
                    return 0.8;
                case 'steady':
                    return 1;
                case 'random':
                    return Math.random() * 2 + 0.5;
                default:
                    return 1;
            }
        }

        function updateCharts() {
            if (charts.traffic && charts.device) {
                // Update traffic chart with real data
                const hours = Array.from({length: 24}, (_, i) => i);
                const newVisitorData = hours.map(hour => {
                    const hourlyVisitors = appState.urls.reduce((sum, url) => {
                        return sum + Math.floor((url.visitors / 24) * (1 + Math.sin(hour * Math.PI / 12) * 0.3));
                    }, 0);
                    return Math.max(hourlyVisitors, Math.floor(Math.random() * 20) + 5);
                });
                
                const newClickData = newVisitorData.map(visitors => Math.floor(visitors * 0.3 * Math.random()) + 2);
                
                charts.traffic.data.datasets[0].data = newVisitorData;
                charts.traffic.data.datasets[1].data = newClickData;
                charts.traffic.update('none');

                // Update device distribution
                const deviceTotals = appState.urls.reduce((acc, url) => {
                    acc.desktop += url.deviceStats.desktop;
                    acc.tablet += url.deviceStats.tablet;
                    acc.android += url.deviceStats.android;
                    acc.iphone += url.deviceStats.iphone;
                    return acc;
                }, { desktop: 0, tablet: 0, android: 0, iphone: 0 });

                if (deviceTotals.desktop + deviceTotals.tablet + deviceTotals.android + deviceTotals.iphone > 0) {
                    charts.device.data.datasets[0].data = [
                        deviceTotals.desktop,
                        deviceTotals.tablet,
                        deviceTotals.android,
                        deviceTotals.iphone
                    ];
                    charts.device.update('none');
                }
            }
        }

        function detectPlatform(url) {
            const platforms = {
                'facebook.com': 'facebook',
                'instagram.com': 'instagram',
                'youtube.com': 'youtube',
                'youtu.be': 'youtube',
                'twitter.com': 'twitter',
                'x.com': 'twitter',

                'linkedin.com': 'linkedin'

            for (const [domain, platform] of Object.entries(platforms)) {
                if (url.includes(domain)) return platform;

            return 'website';
        }

        function getRandomGeoLocation() {
            const random = Math.random() * 100;
            let cumulative = 0;
            
            for (const location of appState.geoLocations) {
                cumulative += location.percentage;
                if (random <= cumulative) {
                    return location;
                }
            }
            
            return appState.geoLocations[appState.geoLocations.length - 1];
        }

                    updateUrlsDisplay();
                    updateAnalytics();
                    showNotification(`✅ AI মাল্টি-ডিভাইস অপ্টিমাইজেশন সম্পন্ন! CVR: ${url.conversionRate}%`, 'success');
                    logActivity(`🧠 AI multi-device optimization completed for ${url.title}`, 'success');
                }, 3000);
            }
        }

        function analyzeUrl(urlId) {
            const url = appState.urls.find(u => u.id == urlId);
            if (url) {
                showNotification(`📊 ${url.title} এর মাল্টি-ডিভাইস গভীর বিশ্লেষণ শুরু হয়েছে`, 'info');
                logActivity(`📊 Multi-device deep analysis started for: ${url.title}`, 'info');
                
                setTimeout(() => {
                    const analysis = generateAnalysisReport(url);
                    showNotification(`✅ মাল্টি-ডিভাইস বিশ্লেষণ সম্পন্ন! ${analysis}`, 'success');
                }, 2000);
            }
        }

        function generateAnalysisReport(url) {
            const reports = [
                `Multi-device high engagement rate detected`,
                `Optimal multi-device conversion potential`,
                `Strong organic growth pattern across all devices`,
                `Excellent multi-device optimization`,
                `Premium multi-device traffic quality confirmed`
            ];
            return reports[Math.floor(Math.random() * reports.length)];
        }

        function bulkOptimize() {
            if (appState.urls.length === 0) {
                showNotification('প্রথমে URL যোগ করুন', 'error');
                return;
            }
            
            showNotification('🧠 সব URL এর জন্য মাল্টি-ডিভাইস Bulk optimization শুরু...', 'info');
            
            appState.urls.forEach((url, index) => {
                setTimeout(() => {
                    url.conversionRate = Math.floor(Math.random() * 25) + 15;
                    url.visitors += Math.floor(Math.random() * 100) + 50;
                    
                    // Distribute across devices
                    ['desktop', 'tablet', 'android', 'iphone'].forEach(device => {
                        url.deviceStats[device] += Math.floor(Math.random() * 30) + 10;
                    });
                    
                    if (index === appState.urls.length - 1) {
                        updateUrlsDisplay();
                        updateAnalytics();
                        showNotification('✅ সব URL এর জন্য মাল্টি-ডিভাইস Bulk optimization সম্পন্ন!', 'success');
                        logActivity('🧠 Multi-device bulk optimization completed for all URLs', 'success');
                    }
                }, index * 1000);
            });
        }

        function logActivity(message, type) {
            const feedContainer = document.getElementById('activityFeed');
            const activityElement = document.createElement('div');
            activityElement.className = 'activity-item fade-in';
            activityElement.innerHTML = `
                <div class="flex items-center gap-2">
                    <div class="w-2 h-2 ${type === 'success' ? 'bg-green-500' : type === 'error' ? 'bg-red-500' : 'bg-blue-500'} rounded-full"></div>
                    <span class="text-xs font-semibold">[${new Date().toLocaleTimeString()}] ${message}</span>
                </div>
            `;

            feedContainer.insertBefore(activityElement, feedContainer.firstChild);

            const activities = feedContainer.children;
            if (activities.length > 50) {
                feedContainer.removeChild(activities[activities.length - 1]);
            }
        }

        function showNotification(message, type) {
            const notification = document.createElement('div');
            notification.className = `notification ${type}`;
            notification.innerHTML = `
                <i class="fas ${type === 'success' ? 'fa-check-circle' : type === 'error' ? 'fa-exclamation-circle' : 'fa-info-circle'} mr-2"></i>
                ${message}
            `;
            
            document.body.appendChild(notification);
            
            setTimeout(() => notification.classList.add('show'), 100);
            setTimeout(() => {
                notification.classList.remove('show');
                setTimeout(() => {
                    if (notification.parentNode) {
                        notification.parentNode.removeChild(notification);
                    }
                }, 300);
            }, 4000);
        }
    </script>
